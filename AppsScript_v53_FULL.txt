function doGet(e) {
  const _ins = doGet_insights(e); if (_ins) return _ins;
  const p = (e && e.parameter) || {};

  // ── Authentication (unauthenticated endpoints) ──
  // Login — username/password from query string, returns token
  if (p.action === 'login') {
    return json(handleLogin_(p.username, p.password));
  }

  // Logout — invalidate a token
  if (p.action === 'logout') {
    return json(handleLogout_(p.token));
  }

  // Session check — dashboard calls this on load to verify stored token
  if (p.action === 'whoami') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    return json({ ok: true, username: auth.username, school: auth.school, role: auth.role });
  }

  // ── Authenticated endpoints from here on ──

  // Teacher dashboard: school identity card for About tab
  if (p.action === 'about_school') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    const deviceId = (p.device_id || '').trim().toUpperCase();
    if (!deviceId) return json({ ok: false, error: 'missing_device_id' });
    return json(buildAboutSchool(deviceId));
  }

  // Teacher dashboard: aggregated classroom data for a school
  if (p.action === 'dashboard_data') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    const school = (p.school || '').trim();
    if (!school) return json({ ok: false, error: 'missing_school', detail: 'Provide ?school=SchoolName' });
    // Enforce scope: non-admin tokens can only read their own school
    if (auth.role !== 'admin' && auth.school !== school) {
      return json({ ok: false, error: 'forbidden', detail: 'Token not authorised for this school' });
    }
    // v50.1: CacheService — first call rebuilds (slow), subsequent calls within
    // 60s serve from cache (~50ms). Each school has its own cache key. Cache is
    // invalidated when telemetry POSTs land (doPost) or when admin updates a
    // device. 60s feels live to teachers and slashes load on the script.
    const cache = CacheService.getScriptCache();
    const cacheKey = 'dash:' + sanitizeSheetName(school);
    const cached = cache.get(cacheKey);
    if (cached) {
      // Stamp the cached payload with a flag so we can see hits in the network tab
      try {
        const obj = JSON.parse(cached);
        obj.cached = true;
        return json(obj);
      } catch (e) {
        // Cache corrupted, fall through and rebuild
      }
    }
    const fresh = buildDashboardData(school);
    try {
      // CacheService values capped at 100KB — most school payloads will fit.
      // If a school's data exceeds that, JSON.stringify will produce something
      // longer than 100KB and put() throws — we catch and skip caching for that
      // school. Functionally equivalent to no cache for huge schools.
      const serialised = JSON.stringify(fresh);
      // v52: Warn when payload approaches CacheService's silent 100KB cap, so
      // future cache-misses for big schools are diagnosable.
      if (serialised.length > 80000) {
        Logger.log('CACHE WARNING: school=%s size=%d (approaching 100KB cap)', school, serialised.length);
      }
      if (serialised.length < 95000) {
        cache.put(cacheKey, serialised, 60);  // 60-second TTL
      }
    } catch (e) {
      // Cache write failed — log and serve uncached
      console.warn('Dashboard cache write skipped:', e);
    }
    return json(fresh);
  }

  // Teacher dashboard: real day-level data for the Daily Report page
  // Called via: ?action=daily_data&school=School+Name&date=YYYY-MM-DD
  if (p.action === 'daily_data') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    const school = (p.school || '').trim();
    const dateStr = (p.date || '').trim();
    if (!school) return json({ ok: false, error: 'missing_school', detail: 'Provide ?school=SchoolName' });
    if (!dateStr || !/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) {
      return json({ ok: false, error: 'missing_or_invalid_date', detail: 'Provide ?date=YYYY-MM-DD' });
    }
    if (auth.role !== 'admin' && auth.school !== school) {
      return json({ ok: false, error: 'forbidden', detail: 'Token not authorised for this school' });
    }
    return json(buildDailyData(school, dateStr));
  }

  // v42: Teacher's own feedback for today (for the dashboard widget to know
  // whether to show "rate today" or "you've already rated").
  if (p.action === 'my_feedback_today') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    return json(getMyFeedbackToday_(auth));
  }

  // v42: Aggregated feedback summary — admin only, never returns raw teacher
  // names or comments to non-admins. SLT explicitly excluded by design.
  if (p.action === 'feedback_summary') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    if (auth.role !== 'admin') {
      return json({ ok: false, error: 'forbidden', detail: 'Admin role required' });
    }
    const school = (p.school || '').trim();   // optional filter
    const days   = Math.min(Number(p.days || 30), 180); // 30 days default, 6 months max
    return json(getFeedbackSummary_(school, days));
  }

  // v43: Caller's own past reviews (for the history list on the Reviews page)
  if (p.action === 'my_reviews') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    const limit = Math.min(Number(p.limit || 20), 100);
    return json(getMyReviews_(auth, limit));
  }

  // v43: Aggregated reviews — admin only. Like feedback_summary but for the
  // longer-form reviews. Returns named comments only to admin role.
  if (p.action === 'reviews_summary') {
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    if (auth.role !== 'admin') {
      return json({ ok: false, error: 'forbidden', detail: 'Admin role required' });
    }
    const school = (p.school || '').trim();
    const days   = Math.min(Number(p.days || 90), 365);
    return json(getReviewsSummary_(school, days));
  }

  // Admin: list all devices
  // Admin endpoints accept either the old ADMIN_KEY (backwards compat for emergencies)
  // OR a valid admin-role token from the login system.
  if (p.action === 'list_devices') {
    if (p.admin_key === ADMIN_KEY) return json(listAllDevices());
    const auth = validateToken_(p.token);
    if (!auth.ok) return json(auth);
    if (auth.role !== 'admin') return json({ ok: false, error: 'forbidden', detail: 'Admin role required' });
    return json(listAllDevices());
  }

  // v45: Read device event history (for admin dashboard timeline)
  if (p.action === 'device_events') {
    return json(getDeviceEvents_(p.device_id, p.limit));
  }

  // v49: Manual fleet mic offset recompute (admin only). Use this to force
  // a refresh outside the rate limit (default 10 min per school). The
  // result includes per-device decisions so you can verify on the dashboard.
  if (p.action === 'compute_mic_offsets') {
    return json(handleComputeMicOffsets_(p.token));
  }

  if (p.selftest === '1') {
    const school = p.school || 'SELFTEST';
    const sheetName = sanitizeSheetName(school);
    const ctx = ensureSheetAndHeaders(sheetName);
    const now = new Date();
    const row = buildRow(ctx.headers, { timestamp: now });
    ctx.sheet.appendRow(row);
    return json({ ok: true, message: 'Self-test row appended', sheet: sheetName });
  }

  return json({ ok: true, message: 'Receiver ready (v45)' });
}

function doPost(e) {
  const _ins = doPost_insights(e); if (_ins) return _ins;
  const rec = parseIncoming(e);

  // v41: Support ticket — emails hello@classroomsounds.co.uk via MailApp.
  // Any authenticated user (staff/slt/admin) can submit.
  if (rec.data.action === 'submit_ticket') {
    return json(handleSubmitTicket_(rec.data));
  }

  // v42: Daily feedback submission — staff only (SLT/admin can submit
  // but they don't have a classroom so it's pointless for them).
  // One submission per teacher per day, editable until midnight.
  if (rec.data.action === 'submit_daily_feedback') {
    return json(handleSubmitDailyFeedback_(rec.data));
  }

  // v43: Full review submission — any authenticated role. Submit any time,
  // any number of times. Append-only.
  if (rec.data.action === 'submit_full_review') {
    return json(handleSubmitFullReview_(rec.data));
  }

  // v45: Device milestone events from Holly v5071+ firmware
  if (rec.data.action === 'device_event') {
    return json(handleDeviceEvent_(rec.data));
  }

  // ── Admin actions (from dashboard) ──
  // Accept either the old ADMIN_KEY or a valid admin-role token.
  if (rec.data.admin_action) {
    var adminAuthorised = false;
    if (rec.data.admin_key === ADMIN_KEY) {
      adminAuthorised = true;
    } else if (rec.data.token) {
      const auth = validateToken_(rec.data.token);
      if (auth.ok && auth.role === 'admin') adminAuthorised = true;
    }
    if (!adminAuthorised) {
      return json({ ok: false, error: 'auth_required', detail: 'Admin token or key required' });
    }
    const action = rec.data.admin_action;
    if (action === 'register_device')  return json(adminRegisterDevice(rec.data));
    if (action === 'update_device')    return json(adminUpdateDevice(rec.data));
    if (action === 'delete_device')    return json(adminDeleteDevice(rec.data));
    return json({ ok: false, error: 'unknown_admin_action', detail: action });
  }

  // ── Device telemetry (from Holly firmware) ──
  logRaw(rec);

  const deviceId = ((rec.data.device_id || rec.data.Device_ID || '')).toString().trim().toUpperCase();

  if (!deviceId) {
    return json({ ok: false, error: 'missing_device_id', detail: 'No device_id in payload' });
  }

  const registry = lookupDevice(deviceId);

  if (!registry) {
    return json({ ok: false, error: 'unregistered_device', detail: 'Device ID ' + deviceId + ' not found in registry' });
  }

  const registrySchool = registry.schoolName;
  const sheetName = sanitizeSheetName(registrySchool);

  rec.data._registrySchool = registrySchool;

  const ctx = ensureSheetAndHeaders(sheetName);

  const lock = LockService.getPublicLock();
  lock.waitLock(30000);
  let appended = false;
  try {
    // v50: Heartbeats are health-only payloads. They update the registry
    // but should not pollute the school data tab — that tab is for lesson
    // telemetry only. Without this filter the tab grows ~20x faster and
    // dashboard reads time out under load.
    const evType = String(rec.data.event_type || '').toUpperCase();
    const isHeartbeat = (evType === 'HEARTBEAT');

    // Belt-and-braces: device_event payloads are handled earlier in doPost
    // (line ~541) and should never reach this code path. If somehow they
    // do — don't write them to the school tab.
    const isDeviceEvent = (rec.data.action === 'device_event');

    if (!isHeartbeat && !isDeviceEvent) {
      const row = buildRow(ctx.headers, rec.data);

      const ignoreIdx = ['Date','Time','DateTime']
        .map(h => ctx.headers.indexOf(h))
        .filter(i => i >= 0);
      const nonBlank = row.filter((v,i) => !ignoreIdx.includes(i) && String(v).trim() !== '');
      if (nonBlank.length > 0) {
        ctx.sheet.appendRow(row);
        appended = true;
      }
    }

    // Update registry with device health (always — this is what heartbeats
    // were always for in the first place)
    updateDeviceHealth(registry.rowIndex, rec.data);

  } catch (err) {
    console.error('append error', err);
    return json({ ok: false, error: 'append_failed', detail: String(err), to: sheetName });
  } finally {
    lock.releaseLock();
  }

  // v52: Run any queued fleet mic offset compute OUTSIDE the lock. The
  // 60s rate-limit was already enforced inside updateDeviceHealth before
  // the flag was set, so reading + clearing here is safe and bounded.
  // We don't await it under any other lock — computeFleetMicOffsets_ takes
  // its own writes against the registry but since we're outside the public
  // lock, other Hollies' POSTs are no longer blocked by it.
  try {
    const _props = PropertiesService.getScriptProperties();
    const _pendingSchool = _props.getProperty('pendingMicOffsetSchool');
    if (_pendingSchool) {
      _props.deleteProperty('pendingMicOffsetSchool');
      console.log('[doPost] Running deferred computeFleetMicOffsets_ triggered by school: ' + _pendingSchool);
      try {
        computeFleetMicOffsets_({ force: false, dryRun: false });
      } catch (e) {
        console.error('Deferred computeFleetMicOffsets_ failed:', e);
      }
    }
  } catch (e) {
    console.error('Pending mic offset check failed:', e);
  }

  const config = buildConfigResponse(registry);

  // v50.1: Invalidate the dashboard cache for this school if real telemetry
  // landed. Heartbeats don't append (so don't invalidate), event_event payloads
  // don't append either. Only real lesson telemetry busts the cache. Means the
  // 60-second TTL is the worst-case staleness — and only for HEARTBEAT-only
  // periods, when there's nothing new to show anyway.
  if (appended) {
    try {
      CacheService.getScriptCache().remove('dash:' + sheetName);
    } catch (e) {
      // Non-fatal — cache will expire on its own in 60s
    }
  }

  return json({
    ok: true,
    appended,
    to: sheetName,
    config: config
  });
}

/* ==================== AUTH (login / tokens) ==================== */
// Pilot-grade auth. Plain-text passwords in a 'Logins' tab; random tokens
// stored in a 'Sessions' tab with an Expires timestamp. When moving to
// Supabase in Phase 2, this whole section gets replaced with Supabase auth
// and the existing endpoints' validateToken_() call sites stay unchanged —
// just swap this one function's implementation.

const AUTH_LOGINS_TAB  = 'Logins';
const AUTH_SESSIONS_TAB = 'Sessions';
const AUTH_TOKEN_TTL_HOURS = 24;

function handleLogin_(username, password) {
  username = String(username || '').trim();
  password = String(password || '');
  if (!username || !password) {
    return { ok: false, error: 'missing_credentials', detail: 'Username and password required' };
  }

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(AUTH_LOGINS_TAB);
  if (!sh) return { ok: false, error: 'auth_not_configured', detail: 'Logins tab missing' };
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return { ok: false, error: 'no_accounts', detail: 'No logins configured' };

  // Columns: Username | Password | School | Role | Active | Created | Last Login
  const data = sh.getRange(2, 1, lastRow - 1, 7).getValues();
  var matchRow = -1, matchUser = null;
  for (var i = 0; i < data.length; i++) {
    const u = String(data[i][0] || '').trim();
    if (u !== username) continue;
    const storedPw = String(data[i][1] || '');
    if (storedPw !== password) continue;
    // Active check — accept TRUE, 'TRUE', 'true', or the boolean
    const active = data[i][4];
    const isActive = active === true || String(active).toLowerCase() === 'true';
    if (!isActive) continue;
    matchRow = i + 2; // +2 for header row and 1-indexed
    matchUser = {
      username: u,
      school:   String(data[i][2] || '').trim(),
      role:     String(data[i][3] || 'staff').trim().toLowerCase()
    };
    break;
  }

  if (matchRow < 0) {
    // Uniform error — don't leak whether the username exists
    return { ok: false, error: 'invalid_credentials', detail: 'Wrong username or password' };
  }

  // Issue token
  const token = generateSessionToken_();
  const now = new Date();
  const expires = new Date(now.getTime() + AUTH_TOKEN_TTL_HOURS * 60 * 60 * 1000);

  // Write session row
  const sessSh = ss.getSheetByName(AUTH_SESSIONS_TAB);
  if (!sessSh) return { ok: false, error: 'auth_not_configured', detail: 'Sessions tab missing' };
  sessSh.appendRow([
    token,
    matchUser.username,
    matchUser.school,
    matchUser.role,
    now,
    expires,
    now
  ]);

  // Update Last Login in Logins tab
  sh.getRange(matchRow, 7).setValue(now);

  return {
    ok: true,
    token: token,
    username: matchUser.username,
    school: matchUser.school,
    role: matchUser.role,
    expires: expires.toISOString()
  };
}

function handleLogout_(token) {
  token = String(token || '').trim();
  if (!token) return { ok: false, error: 'missing_token' };
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(AUTH_SESSIONS_TAB);
  if (!sh) return { ok: true }; // nothing to delete — treat as success
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return { ok: true };
  const tokens = sh.getRange(2, 1, lastRow - 1, 1).getValues();
  for (var i = 0; i < tokens.length; i++) {
    if (String(tokens[i][0]) === token) {
      sh.deleteRow(i + 2);
      return { ok: true, message: 'Logged out' };
    }
  }
  return { ok: true }; // token not found — treat as already logged out
}

/**
 * Validate a session token. Returns { ok:true, username, school, role } on
 * success or { ok:false, error:'auth_required'|'invalid_token'|'expired_token' }.
 * Called at the top of every protected endpoint.
 */
function validateToken_(token) {
  token = String(token || '').trim();
  if (!token) return { ok: false, error: 'auth_required', detail: 'No token provided' };

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(AUTH_SESSIONS_TAB);
  if (!sh) return { ok: false, error: 'auth_not_configured', detail: 'Sessions tab missing' };
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return { ok: false, error: 'invalid_token' };

  // Columns: Token | Username | School | Role | Created | Expires | Last Used
  const data = sh.getRange(2, 1, lastRow - 1, 7).getValues();
  const now = new Date();
  for (var i = 0; i < data.length; i++) {
    if (String(data[i][0]) !== token) continue;
    var expires = data[i][5];
    if (!(expires instanceof Date)) expires = new Date(expires);
    if (isNaN(expires.getTime()) || expires < now) {
      // Expired — remove the row
      sh.deleteRow(i + 2);
      return { ok: false, error: 'expired_token', detail: 'Session expired — please log in again' };
    }
    // Update Last Used (fire-and-forget; don't fail auth if it errors)
    try { sh.getRange(i + 2, 7).setValue(now); } catch (e) { /* ignore */ }
    return {
      ok: true,
      username: String(data[i][1] || ''),
      school:   String(data[i][2] || ''),
      role:     String(data[i][3] || 'staff').toLowerCase()
    };
  }
  return { ok: false, error: 'invalid_token', detail: 'Token not recognised — please log in again' };
}

function generateSessionToken_() {
  // 32 hex chars = 128 bits of entropy. Good enough for pilot.
  // Apps Script has no crypto.randomUUID so we build from Math.random + timestamp.
  var s = '';
  for (var i = 0; i < 4; i++) {
    s += Math.floor(Math.random() * 0xffffffff).toString(16).padStart(8, '0');
  }
  return s;
}

/**
 * Admin helper — wipe all expired sessions. Safe to run anytime.
 * Hook this up to a time-based trigger (daily) if you want automatic cleanup,
 * otherwise expired tokens naturally get deleted when validateToken_ trips on them.
 */
function cleanupExpiredSessions() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sh = ss.getSheetByName(AUTH_SESSIONS_TAB);
  if (!sh) return;
  const lastRow = sh.getLastRow();
  if (lastRow < 2) return;
  const data = sh.getRange(2, 1, lastRow - 1, 7).getValues();
  const now = new Date();
  var removed = 0;
  // Delete from bottom up so row indices don't shift
  for (var i = data.length - 1; i >= 0; i--) {
    var expires = data[i][5];
    if (!(expires instanceof Date)) expires = new Date(expires);
    if (isNaN(expires.getTime()) || expires < now) {
      sh.deleteRow(i + 2);
      removed++;
    }
  }
  console.log('cleanupExpiredSessions: removed ' + removed + ' expired sessions');
}

function buildConfigResponse(registry) {
  const cfg = {};
  cfg.config_version = registry.config.config_version || 0;

  if (registry.config.set_year_group !== '')  cfg.set_year_group  = registry.config.set_year_group;

  if (registry.config.set_break_start !== '') cfg.set_break_start = registry.config.set_break_start;
  if (registry.config.set_break_end !== '')   cfg.set_break_end   = registry.config.set_break_end;

  if (registry.config.set_day_end !== '')     cfg.set_day_end     = registry.config.set_day_end;
  // v37.3: Afternoon break — always include (even blank) so Holly can
  // clear the field if admin removes it. The other timetable fields guard
  // on non-empty because they're required; pmbreak is the only one where
  // "" is a meaningful "disabled" value rather than "no change requested".
  cfg.set_pmbreak_start = registry.config.set_pmbreak_start || '';
  cfg.set_pmbreak_end   = registry.config.set_pmbreak_end   || '';

  // Reboot command — send once then auto-clear
  if (registry.config.reboot_requested === true) {
    cfg.reboot = true;
    try {
      const ss = SpreadsheetApp.openById(SHEET_ID);
      const regSheet = ss.getSheetByName(REGISTRY_SHEET);
      if (regSheet) {
        const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
        const rebootCol = headers.findIndex(function(h) { return String(h).trim() === 'Reboot Requested'; });
        if (rebootCol >= 0) regSheet.getRange(registry.rowIndex, rebootCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear reboot flag', e); }
  }

  // Room test command — send once then auto-clear (v5026)
  if (registry.config.room_test_requested === true) {
    cfg.room_test = true;
    try {
      const ss2 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet2 = ss2.getSheetByName(REGISTRY_SHEET);
      if (regSheet2) {
        const headers2 = regSheet2.getRange(1, 1, 1, regSheet2.getLastColumn()).getValues()[0];
        const rtCol = headers2.findIndex(function(h) { return String(h).trim() === 'Room Test Requested'; });
        if (rtCol >= 0) regSheet2.getRange(registry.rowIndex, rtCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear room test flag', e); }
  }

  // v5037: Locate command ("Find My Holly") — send once then auto-clear.
  // Values: 'true'  -> locate (60s magenta/cyan flash on device)
  //         'cancel' -> cancel an in-progress locate
  //         empty/other -> no-op
  if (registry.config.locate_requested === 'true') {
    cfg.locate = true;
    try {
      const ss3 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet3 = ss3.getSheetByName(REGISTRY_SHEET);
      if (regSheet3) {
        const headers3 = regSheet3.getRange(1, 1, 1, regSheet3.getLastColumn()).getValues()[0];
        const locCol = headers3.findIndex(function(h) { return String(h).trim() === 'Locate Requested'; });
        if (locCol >= 0) regSheet3.getRange(registry.rowIndex, locCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear locate flag', e); }
  } else if (registry.config.locate_requested === 'cancel') {
    cfg.locate_cancel = true;
    try {
      const ss4 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet4 = ss4.getSheetByName(REGISTRY_SHEET);
      if (regSheet4) {
        const headers4 = regSheet4.getRange(1, 1, 1, regSheet4.getLastColumn()).getValues()[0];
        const locCol4 = headers4.findIndex(function(h) { return String(h).trim() === 'Locate Requested'; });
        if (locCol4 >= 0) regSheet4.getRange(registry.rowIndex, locCol4 + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear locate cancel flag', e); }
  }

  // v37 / firmware v5042: Self-test command — send once, auto-clear the
  // Set Self Test flag in the registry. Results arrive back in the NEXT
  // telemetry POST as self_test_* keys, written by updateDeviceHealth().
  if (registry.config.set_self_test === true) {
    cfg.self_test = true;
    try {
      const ss5 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet5 = ss5.getSheetByName(REGISTRY_SHEET);
      if (regSheet5) {
        const headers5 = regSheet5.getRange(1, 1, 1, regSheet5.getLastColumn()).getValues()[0];
        const stCol = headers5.findIndex(function(h) { return String(h).trim() === 'Set Self Test'; });
        if (stCol >= 0) regSheet5.getRange(registry.rowIndex, stCol + 1).clearContent();
        // Mark that a test is in flight so the dashboard can show a
        // spinner until results arrive. The status cell is cleared and
        // repopulated by updateDeviceHealth on the next POST.
        const statusCol = headers5.findIndex(function(h) { return String(h).trim() === 'Self Test Status'; });
        if (statusCol >= 0) regSheet5.getRange(registry.rowIndex, statusCol + 1).setValue('running');
      }
    } catch (e) { console.error('Failed to clear self-test flag', e); }
  }

  // v46 / firmware v5073: Remote learning-mode trigger.
  // Sends set_room_cal_enabled:true to the device, which always resets
  // calibration state cleanly and starts a fresh 14-day Welford window.
  // Auto-clears the column after one delivery (one-shot, same pattern
  // as Reboot/Room Test/Locate).
  if (registry.config.set_room_cal_enabled === true) {
    cfg.set_room_cal_enabled = true;
    try {
      const ss6 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet6 = ss6.getSheetByName(REGISTRY_SHEET);
      if (regSheet6) {
        const headers6 = regSheet6.getRange(1, 1, 1, regSheet6.getLastColumn()).getValues()[0];
        const rcCol = headers6.findIndex(function(h) { return String(h).trim() === 'Set Room Cal Enabled'; });
        if (rcCol >= 0) regSheet6.getRange(registry.rowIndex, rcCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear set_room_cal_enabled flag', e); }
  }

  // v47 / firmware v5074: Remote learning CANCEL.
  // Sends cancel_learning:true to the device, which does a hard cancel
  // (disables learning, wipes Welford counters, reverts to band defaults).
  // Auto-clears the column after one delivery.
  if (registry.config.cancel_learning === true) {
    cfg.cancel_learning = true;
    try {
      const ss7 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet7 = ss7.getSheetByName(REGISTRY_SHEET);
      if (regSheet7) {
        const headers7 = regSheet7.getRange(1, 1, 1, regSheet7.getLastColumn()).getValues()[0];
        const clCol = headers7.findIndex(function(h) { return String(h).trim() === 'Cancel Learning'; });
        if (clCol >= 0) regSheet7.getRange(registry.rowIndex, clCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear cancel_learning flag', e); }
  }

  // v49: Per-unit mic offset correction. Delivered as a numeric value
  // (positive or negative). Auto-cleared after delivery so the same
  // correction isn't re-pushed every check-in. Mirrors the v5075 one-shot
  // pattern used for set_room_cal_enabled and cancel_learning.
  const setOffset = registry.config.set_mic_db_offset_correction;
  if (setOffset !== undefined && setOffset !== null && setOffset !== '') {
    const offsetNum = Number(setOffset);
    if (isFinite(offsetNum) && Math.abs(offsetNum) > 0.05) {
      cfg.mic_db_offset_correction = offsetNum;
      try {
        const ss8 = SpreadsheetApp.openById(SHEET_ID);
        const regSheet8 = ss8.getSheetByName(REGISTRY_SHEET);
        if (regSheet8) {
          const headers8 = regSheet8.getRange(1, 1, 1, regSheet8.getLastColumn()).getValues()[0];
          const moCol = headers8.findIndex(function(h) { return String(h).trim() === 'Set Mic DB Offset Correction'; });
          if (moCol >= 0) regSheet8.getRange(registry.rowIndex, moCol + 1).clearContent();
        }
      } catch (e) { console.error('Failed to clear set_mic_db_offset_correction flag', e); }
    }
  }

  // v49 (Option B): Clear Holly's firmware-side lastError buffer remotely.
  // Set by the dashboard when admin clicks "Clear Error". Apps Script
  // delivers clear_last_error:true in the config response. Holly v5078+
  // receives it and zeroes her lastError[64] buffer. Auto-clears the
  // registry cell so it's a one-shot event.
  if (registry.config.clear_last_error === true) {
    cfg.clear_last_error = true;
    try {
      const ss9 = SpreadsheetApp.openById(SHEET_ID);
      const regSheet9 = ss9.getSheetByName(REGISTRY_SHEET);
      if (regSheet9) {
        const headers9 = regSheet9.getRange(1, 1, 1, regSheet9.getLastColumn()).getValues()[0];
        const cleCol = headers9.findIndex(function(h) { return String(h).trim() === 'Clear Last Error'; });
        if (cleCol >= 0) regSheet9.getRange(registry.rowIndex, cleCol + 1).clearContent();
      }
    } catch (e) { console.error('Failed to clear clear_last_error flag', e); }
  }

  return cfg;
}

/* ==================== REGISTRY FUNCTIONS ==================== */

function lookupDevice(deviceId) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let regSheet = ss.getSheetByName(REGISTRY_SHEET);

  if (!regSheet) {
    regSheet = ss.insertSheet(REGISTRY_SHEET);
    regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setValues([REGISTRY_HEADERS]);
    regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setFontWeight('bold');
    return null;
  }

  const lastRow = regSheet.getLastRow();
  if (lastRow < 2) return null;

  const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
  const data = regSheet.getRange(2, 1, lastRow - 1, headers.length).getValues();

  const col = {};
  headers.forEach((h, i) => { col[String(h).trim()] = i; });

  for (let i = 0; i < data.length; i++) {
    const rowDeviceId = String(data[i][col['Device ID']] || '').trim().toUpperCase();
    if (rowDeviceId === deviceId) {
      return {
        schoolName: String(data[i][col['School Name']] || '').trim(),
        rowIndex: i + 2,
        colIndex: col,
        config: {
          config_version:    numOrEmpty(data[i][col['Config Version']]),
          set_year_group:    strOrEmpty(data[i][col['Set Year Group']]),
          set_classroom:     strOrEmpty(data[i][col['Set Classroom']]),
          set_green_max:     numOrEmpty(data[i][col['Set Green Max']]),
          set_amber_max:     numOrEmpty(data[i][col['Set Amber Max']]),
          set_red_warn:      numOrEmpty(data[i][col['Set Red Warn']]),
          set_brightness:    numOrEmpty(data[i][col['Set Brightness']]),
          set_timetable_enabled: boolOrEmpty(data[i][col['Set Timetable Enabled']]),
          set_school_open:     boolOrEmpty(data[i][col['Set School Open']]),
          set_reopen_date:     strOrEmpty(data[i][col['Set Reopen Date']]),
          manual_override:   boolOrEmpty(data[i][col['Manual Override']]),
          set_send_count:    numOrEmpty(data[i][col['Set SEND Count']]),
          sensitivity_profile: strOrEmpty(data[i][col['Sensitivity Profile']]),
          set_day_start:     strOrEmpty(data[i][col['Set Day Start']]),
          set_break_start:   strOrEmpty(data[i][col['Set Break Start']]),
          set_break_end:     strOrEmpty(data[i][col['Set Break End']]),
          set_lunch_start:   strOrEmpty(data[i][col['Set Lunch Start']]),
          set_lunch_end:     strOrEmpty(data[i][col['Set Lunch End']]),
          set_day_end:       strOrEmpty(data[i][col['Set Day End']]),
          set_pmbreak_start: strOrEmpty(data[i][col['Set Afternoon Break Start']]),
          set_pmbreak_end:   strOrEmpty(data[i][col['Set Afternoon Break End']]),
          reboot_requested:  boolOrEmpty(data[i][col['Reboot Requested']]),
          room_test_requested: boolOrEmpty(data[i][col['Room Test Requested']]),
          locate_requested:  strOrEmpty(data[i][col['Locate Requested']]),
          set_self_test:     boolOrEmpty(data[i][col['Set Self Test']]),
          set_room_cal_enabled: boolOrEmpty(data[i][col['Set Room Cal Enabled']]),
          cancel_learning:   boolOrEmpty(data[i][col['Cancel Learning']]),
          // v49: Per-unit mic offset correction (numeric, may be negative)
          set_mic_db_offset_correction: numOrEmpty(data[i][col['Set Mic DB Offset Correction']]),
          // v49 (Option B): one-shot clear of Holly's lastError buffer
          clear_last_error:  boolOrEmpty(data[i][col['Clear Last Error']])
        }
      };
    }
  }

  return null;
}

/**
 * v53: Idempotently ensures the Device Registry has the given column
 * headers. Adds any missing columns to the right of the existing ones.
 * Safe to call on every doPost — only writes if a header is missing.
 *
 * @param {string[]} requiredHeaders - column names that must exist
 * @return {Object} a {name: columnIndex} lookup for all headers (1-based)
 */
function ensureRegistryColumns_(requiredHeaders) {
  const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('Device Registry');
  if (!sheet) return {};
  const lastCol = Math.max(1, sheet.getLastColumn());
  const headerRange = sheet.getRange(1, 1, 1, lastCol);
  const headers = headerRange.getValues()[0];
  const lookup = {};
  headers.forEach((h, i) => { if (h) lookup[h] = i + 1; });
  let nextCol = lastCol + 1;
  let added = false;
  for (const name of requiredHeaders) {
    if (!lookup[name]) {
      sheet.getRange(1, nextCol).setValue(name);
      lookup[name] = nextCol;
      nextCol++;
      added = true;
    }
  }
  if (added) SpreadsheetApp.flush();
  return lookup;
}

function updateDeviceHealth(rowIndex, data) {
  try {
    // v53: Ensure new firmware v5083+ columns exist before mapping fields.
    ensureRegistryColumns_([
      'Active Device Phase',
      'Active Window State'
    ]);

    const ss = SpreadsheetApp.openById(SHEET_ID);
    const regSheet = ss.getSheetByName(REGISTRY_SHEET);
    if (!regSheet) return;

    const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
    const col = {};
    headers.forEach((h, i) => { col[String(h).trim()] = i; });

    const setCell = (colName, value) => {
      const c = col[colName];
      if (c !== undefined && value !== undefined && value !== null && value !== '') {
        regSheet.getRange(rowIndex, c + 1).setValue(value);
      }
    };

    // Always update Last Seen
    const lsCol = col['Last Seen'];
    if (lsCol !== undefined) {
      regSheet.getRange(rowIndex, lsCol + 1).setValue(new Date());
      regSheet.getRange(rowIndex, lsCol + 1).setNumberFormat('yyyy-mm-dd HH:mm:ss');
    }

    // Device-reported active config
    setCell('Confirmed Version',  data.confirmed_version);
    setCell('Active Green',       data.active_green);
    setCell('Active Amber',       data.active_amber);
    setCell('Active Red',         data.active_red);
    setCell('Active Brightness',  data.active_brightness);
    setCell('Active Year Group',  data.active_year_group);
    setCell('Active Year Band',   data.active_year_band);
    setCell('Active Classroom',   data.active_classroom);

    // v53: device_phase and ws (new in firmware v5083+).
    // data.device_phase / data.ws may be undefined for pre-5083 firmware —
    // write empty string in that case.
    if (col['Active Device Phase'] !== undefined) {
      regSheet.getRange(rowIndex, col['Active Device Phase'] + 1)
              .setValue(data.device_phase || '');
    }
    if (col['Active Window State'] !== undefined) {
      regSheet.getRange(rowIndex, col['Active Window State'] + 1)
              .setValue(data.ws || '');
    }

    setCell('Last Zone',          data.last_zone);
    setCell('Firmware Version',   data.firmware_version);
    setCell('IP Address',         data.ip_address);
    setCell('MAC Address',        data.mac_address);
    setCell('WiFi RSSI',          data.wifi_rssi);
    setCell('Uptime Minutes',     data.uptime_minutes);
    setCell('Last Reset Reason',  data.last_reset_reason);
    // v49 (Option C): Stickiness suppression. If admin recently cleared
    // a specific error string for this device, and Holly's firmware is
    // still reporting that same string from its in-RAM lastError[] buffer,
    // suppress the write so the dashboard alert doesn't re-appear. The
    // stickiness entry has a 24-hour TTL (long enough for Option B
    // firmware to OTA out and clear Holly's buffer, short enough that
    // genuinely new identical errors will eventually be surfaced).
    let suppressLastError = false;
    if (data.last_error && String(data.last_error).trim().length > 0) {
      try {
        const props = PropertiesService.getScriptProperties();
        const stickKey = 'cleared_err_' + (data.device_id ? String(data.device_id).trim().toUpperCase() : '');
        const stickRaw = props.getProperty(stickKey);
        if (stickRaw) {
          const stick = JSON.parse(stickRaw);
          const ageMs = Date.now() - Number(stick.ts || 0);
          if (ageMs < 86400000 && String(stick.err).trim() === String(data.last_error).trim()) {
            // Same string admin cleared, still within 24h — suppress
            suppressLastError = true;
            console.log('[updateDeviceHealth] Suppressed stale lastError for ' + data.device_id + ': ' + data.last_error);
          } else if (ageMs >= 86400000) {
            // TTL expired — clean up
            props.deleteProperty(stickKey);
          }
        }
      } catch (e) {
        console.error('lastError stickiness check failed:', e);
      }
    }
    if (!suppressLastError) {
      setCell('Last Error',         data.last_error);
    }

    // Holly Name (device-reported identity)
    setCell('Holly Name',         data.holly_name);

    // Last temperature
    setCell('Last Temperature',   data.temperature_C);

    // Voice activity percentage
    if (data.voice_percent !== undefined && data.voice_percent !== null) {
      setCell('Voice %', data.voice_percent);
    }

    // Learning progress
    if (data.learning_progress !== undefined && data.learning_progress !== null) {
      setCell('Learning Progress', data.learning_progress);
    }

    // Timetable enabled state
    setCell('Active Timetable Enabled', data.active_timetable_enabled);
    setCell('Active School Open',      data.active_school_open);

    // SEND monitoring (v16) — report active count and auto-seed Set if blank
    if (data.active_send_count !== undefined && data.active_send_count !== null) {
      setCell('Active SEND Count', data.active_send_count);
      // Auto-seed: if device reports a SEND count but Set column is blank, populate it
      const setSendCol = col['Set SEND Count'];
      if (setSendCol !== undefined) {
        const currentSetValue = regSheet.getRange(rowIndex, setSendCol + 1).getValue();
        if (currentSetValue === '' || currentSetValue === null || currentSetValue === undefined) {
          const reportedCount = Number(data.active_send_count);
          if (!isNaN(reportedCount) && reportedCount > 0) {
            regSheet.getRange(rowIndex, setSendCol + 1).setValue(reportedCount);
          }
        }
      }
    }

    // Auto-clear reopen date if device has auto-reopened
    if (data.auto_reopened === 'Yes') {
      const reopenCol = col['Set Reopen Date'];
      const openCol   = col['Set School Open'];
      if (reopenCol !== undefined) regSheet.getRange(rowIndex, reopenCol + 1).clearContent();
      if (openCol !== undefined)   regSheet.getRange(rowIndex, openCol + 1).setValue(true);
    }

    // Mic floor baseline (v5026)
    // v49: Capture the previous floor value BEFORE overwriting, so we can
    // detect whether the floor actually changed and decide whether to
    // trigger the fleet recompute.
    let previousFloor = 0;
    if (data.mic_floor !== undefined && data.mic_floor !== null && col['Mic Floor'] !== undefined) {
      previousFloor = Number(regSheet.getRange(rowIndex, col['Mic Floor'] + 1).getValue()) || 0;
      setCell('Mic Floor', data.mic_floor);
    }
    if (data.mic_floor_valid !== undefined) {
      setCell('Mic Floor Valid', data.mic_floor_valid);
    }
    if (data.mic_floor_date !== undefined && data.mic_floor_date !== null && data.mic_floor_date !== '') {
      setCell('Mic Floor Date', data.mic_floor_date);
    }
    // v49: Per-unit mic offset correction reported by Holly v5077+.
    // Captured into 'Active Mic DB Offset Correction' for dashboard
    // transparency. The 'Set' column is cleared after delivery (in
    // buildConfigResponse) — Active reflects what's actually running.
    if (data.mic_offset_corr !== undefined && data.mic_offset_corr !== null) {
      setCell('Active Mic DB Offset Correction', data.mic_offset_corr);
    }
    // v49 (revised): Only trigger fleet offset recompute when the mic floor
    // VALUE has actually changed since last reported. Without this guard, the
    // function fires on every Holly POST (every 15s during fast-cadence boot
    // window × 14 Hollies = 56 invocations/min), which causes Apps Script
    // timeouts because it reads the entire registry on every call.
    //
    // The mic floor only changes after a room test completes — typically
    // once per device per deployment. So this guard reduces invocations
    // from 56/min to ~14 over a fleet's whole lifetime.
    //
    // Plus a global rate limit (60s minimum between runs) as a backstop
    // in case multiple Hollies happen to complete room tests simultaneously.
    if (String(data.mic_floor_valid || '').toLowerCase() === 'yes' &&
        data.mic_floor !== undefined && Number(data.mic_floor) > 0) {
      try {
        const reportedFloor = Number(data.mic_floor);
        const floorChanged = Math.abs(reportedFloor - previousFloor) > 0.5;

        if (floorChanged) {
          // Global rate limit: never run more than once per 60 seconds
          // regardless of which Holly triggered it.
          //
          // v52: Do NOT run computeFleetMicOffsets_ inline. We're inside the
          // doPost LockService.getPublicLock() block here, and the compute
          // pass reads the entire registry + does many setValue calls — easily
          // 5-10s, which blocks every other Holly's POST and pushes them past
          // their 12s HTTP timeout. Instead, set a Script Property flag which
          // doPost will pick up AFTER releaseLock() and run unlocked.
          const props = PropertiesService.getScriptProperties();
          const lastGlobalRun = Number(props.getProperty('mic_offset_global_last_run') || 0);
          if (Date.now() - lastGlobalRun > 60000) {
            props.setProperty('mic_offset_global_last_run', String(Date.now()));
            const triggerSchool = String(data._registrySchool || '').trim();
            props.setProperty('pendingMicOffsetSchool', triggerSchool || '<unknown>');
          }
        }
      } catch (e) {
        console.error('Auto fleet offset compute failed:', e);
      }
    }

    // Active timetable
    setCell('Active Day Start',   data.active_day_start);
    setCell('Active Break Start', data.active_break_start);
    setCell('Active Break End',   data.active_break_end);
    setCell('Active Lunch Start', data.active_lunch_start);
    setCell('Active Lunch End',   data.active_lunch_end);
    setCell('Active Day End',     data.active_day_end);

    // Self-test results (v37 / firmware v5042) — only present when the
    // device just completed a sweep. Device clears its pending flag once
    // the POST is acknowledged, so results only arrive once per run.
    if (data.self_test_status !== undefined && data.self_test_status !== null && data.self_test_status !== '') {
      setCell('Self Test Status',  data.self_test_status);
      setCell('Self Test Date',    data.self_test_date);
      setCell('Test WiFi',         data.self_test_wifi);
      setCell('Test Sheets',       data.self_test_sheets);
      setCell('Test Heap',         data.self_test_heap);
      setCell('Test Watchdog',     data.self_test_watchdog);
      setCell('Test Mic',          data.self_test_mic);
      setCell('Test BME',          data.self_test_bme);
      setCell('Test LED',          data.self_test_led);
      setCell('Test NVS',          data.self_test_nvs);
      setCell('Test Time',         data.self_test_time);
      setCell('Test Button',       data.self_test_button);
      setCell('Test OTA',          data.self_test_ota);
    }

    // Calibration data
    setCell('Cal Status',         data.cal_status);
    setCell('Cal Mean dB',        data.cal_mean);
    setCell('Cal SD',             data.cal_sd);
    setCell('Cal Min dB',         data.cal_min);
    setCell('Cal Max dB',         data.cal_max);
    setCell('Cal Samples',        data.cal_samples);
    setCell('Cal Green',          data.cal_green);
    setCell('Cal Amber',          data.cal_amber);
    setCell('Cal Red',            data.cal_red);
    setCell('Cal Start Date',     data.cal_start_date);
    setCell('Cal Complete Date',  data.cal_complete_date);

  } catch (err) {
    console.error('updateDeviceHealth error', err);
  }
}

/* ==================== onEdit TRIGGER ==================== */

function onRegistryEdit(e) {
  try {
    const sheet = e.source.getActiveSheet();
    if (sheet.getName() !== REGISTRY_SHEET) return;

    const range = e.range;
    const row = range.getRow();
    const col = range.getColumn();

    if (row < 2) return;

    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const editedHeader = String(headers[col - 1] || '').trim();

    if (!CONFIG_COLUMNS.includes(editedHeader)) return;

    // Increment config version
    const cvColIdx = headers.findIndex(h => String(h).trim() === 'Config Version');
    if (cvColIdx < 0) return;

    const currentVersion = Number(sheet.getRange(row, cvColIdx + 1).getValue()) || 0;
    sheet.getRange(row, cvColIdx + 1).setValue(currentVersion + 1);

    console.log('Config version bumped to ' + (currentVersion + 1) + ' for row ' + row + ' (edited: ' + editedHeader + ')');

  } catch (err) {
    console.error('onRegistryEdit error', err);
  }
}

/**
 * List all devices from registry — returns full row data for dashboard.
 */
function listAllDevices() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: true, devices: [] };

  const lastRow = regSheet.getLastRow();
  const lastCol = regSheet.getLastColumn();
  if (lastRow < 2 || lastCol < 1) return { ok: true, devices: [] };

  const headers = regSheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
  const data = regSheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

  const devices = [];
  data.forEach(row => {
    const deviceId = String(row[headers.indexOf('Device ID')] || '').trim();
    if (!deviceId) return; // skip blank rows
    const obj = {};
    headers.forEach((h, i) => {
      // Convert header to camelCase key
      const key = h.replace(/[^a-zA-Z0-9 ]/g, '').trim()
        .split(' ').map((w, j) => j === 0 ? w.toLowerCase() : w.charAt(0).toUpperCase() + w.slice(1).toLowerCase()).join('');
      let val = row[i];
      // Format dates
      if (val instanceof Date) val = Utilities.formatDate(val, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
      if (val === '' || val === null || val === undefined) val = '';
      obj[key] = val;
    });
    devices.push(obj);
  });

  return { ok: true, devices: devices, columns: headers };
}

/**
 * Register a new device in the registry.
 * Expects: device_id, school_name, and optional location/config fields.
 */
function adminRegisterDevice(data) {
  const deviceId = String(data.device_id || '').trim().toUpperCase();
  if (!deviceId || deviceId.length < 4) {
    return { ok: false, error: 'invalid_device_id', detail: 'Device ID must be at least 4 characters' };
  }

  const schoolName = String(data.school_name || '').trim();
  if (!schoolName) {
    return { ok: false, error: 'missing_school_name', detail: 'School name is required' };
  }

  // v45.1: Public lock around the entire read-check-write cycle.
  // Without this, concurrent admin POSTs (bulk import of N devices) can
  // interleave: two executions read the same lastRow, both append, and
  // one write clobbers the other. Same pattern as the telemetry path.
  const lock = LockService.getPublicLock();
  try {
    lock.waitLock(30000);
  } catch (e) {
    return { ok: false, error: 'lock_timeout', detail: 'Could not acquire write lock (busy backend)' };
  }

  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let regSheet = ss.getSheetByName(REGISTRY_SHEET);
    if (!regSheet) {
      regSheet = ss.insertSheet(REGISTRY_SHEET);
      regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setValues([REGISTRY_HEADERS]);
      regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setFontWeight('bold');
    }

    // Check for duplicate
    const lastRow = regSheet.getLastRow();
    if (lastRow >= 2) {
      const existingIds = regSheet.getRange(2, 1, lastRow - 1, 1).getValues().flat()
        .map(id => String(id).trim().toUpperCase());
      if (existingIds.includes(deviceId)) {
        return { ok: false, error: 'duplicate_device', detail: 'Device ' + deviceId + ' already registered' };
      }
    }

    const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0]
      .map(h => String(h).trim());

    // v37.2: Auto-copy school timetable from an existing device at the same
    // school, if this school already has one registered. Only applies when
    // the caller didn't explicitly supply timetable fields (so the setup
    // wizard for the FIRST device at a school still uses wizard values).
    //
    // Why: the "+ Add Device" flow doesn't ask for a timetable — but the new
    // device should run on the same hours as the rest of the school, not on
    // firmware defaults. Without this, each newly-added Holly would fall
    // back to hardcoded defaults until someone remembered to run Save from
    // the Edit School panel.
    const wantedTTFields = ['set_day_start','set_break_start','set_break_end',
                            'set_lunch_start','set_lunch_end','set_day_end',
                            'set_pmbreak_start','set_pmbreak_end',
                            'set_timetable_enabled'];
    const callerSuppliedTT = wantedTTFields.some(k =>
      data[k] !== undefined && data[k] !== '' && data[k] !== null);

    if (!callerSuppliedTT && lastRow >= 2) {
      const rowData = regSheet.getRange(2, 1, lastRow - 1, headers.length).getValues();
      const schoolCol = headers.indexOf('School Name');
      const ttColMap = {
        'set_day_start':          headers.indexOf('Set Day Start'),
        'set_break_start':        headers.indexOf('Set Break Start'),
        'set_break_end':          headers.indexOf('Set Break End'),
        'set_lunch_start':        headers.indexOf('Set Lunch Start'),
        'set_lunch_end':          headers.indexOf('Set Lunch End'),
        'set_day_end':            headers.indexOf('Set Day End'),
        'set_pmbreak_start':      headers.indexOf('Set Afternoon Break Start'),
        'set_pmbreak_end':        headers.indexOf('Set Afternoon Break End'),
        'set_timetable_enabled':  headers.indexOf('Set Timetable Enabled')
      };
      for (let i = 0; i < rowData.length; i++) {
        if (String(rowData[i][schoolCol] || '').trim() === schoolName) {
          // Match! Copy the timetable fields from this row if present.
          for (const key in ttColMap) {
            const c = ttColMap[key];
            if (c >= 0) {
              const val = rowData[i][c];
              if (val !== '' && val !== null && val !== undefined) {
                data[key] = val;
              }
            }
          }
          break;
        }
      }
    }

    const row = headers.map(h => {
      if (h === 'Device ID')       return deviceId;
      if (h === 'School Name')     return schoolName;
      if (h === 'Org Type')        return data.org_type || '';
      if (h === 'Org Name')        return data.org_name || '';
      if (h === 'Org URN')         return data.org_urn || '';
      if (h === 'School Logo')     return data.school_logo || '';
      if (h === 'Teacher Name')    return data.teacher_name || '';
      if (h === 'Holly Name')      return data.holly_name || '';
      if (h === 'School Address')  return data.school_address || '';
      if (h === 'School Town')     return data.school_town || '';
      if (h === 'School County')   return data.school_county || '';
      if (h === 'School Region')   return data.school_region || '';
      if (h === 'School Country')  return data.school_country || '';
      if (h === 'School Postcode') return data.school_postcode || '';
      if (h === 'Set Year Group')  return data.set_year_group || '';
      if (h === 'Set Classroom')   return data.set_classroom || '';
      if (h === 'Set SEND Count')  return data.set_send_count || '';
      // v37.1: Accept timetable fields from the Set Up New School wizard.
      // v37.2: Auto-copied from existing devices at same school (Add Device flow).
      if (h === 'Set Day Start')          return data.set_day_start          || '';
      if (h === 'Set Break Start')        return data.set_break_start        || '';
      if (h === 'Set Break End')          return data.set_break_end          || '';
      if (h === 'Set Lunch Start')        return data.set_lunch_start        || '';
      if (h === 'Set Lunch End')          return data.set_lunch_end          || '';
      if (h === 'Set Day End')            return data.set_day_end            || '';
      // v37.3: Afternoon break (optional — blank = no afternoon break)
      if (h === 'Set Afternoon Break Start')  return data.set_pmbreak_start  || '';
      if (h === 'Set Afternoon Break End')    return data.set_pmbreak_end    || '';
      if (h === 'Set Timetable Enabled')  {
        // v37.2: Match v36 adminUpdateDevice coercion — checkbox cells
        // silently reject raw JS booleans; must write literal "TRUE"/"FALSE".
        let v = data.set_timetable_enabled;
        if (v === undefined || v === null || v === '') return '';
        if (typeof v === 'boolean') return v ? 'TRUE' : 'FALSE';
        return v;
      }
      if (h === 'Config Version')  return 0;
      if (h === 'Status')          return 'active';
      return '';
    });

    regSheet.appendRow(row);
    SpreadsheetApp.flush();  // v45.1: force commit before lock release

    return { ok: true, message: 'Device ' + deviceId + ' registered at ' + schoolName, device_id: deviceId };
  } catch (err) {
    console.error('adminRegisterDevice error', err);
    return { ok: false, error: 'register_failed', detail: String(err) };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Update fields on an existing device.
 * Expects: device_id, plus any fields to update (using registry column names in snake_case).
 *
 * v36 fix: Coerces JS booleans to "TRUE"/"FALSE" strings before writing.
 * Checkbox-validated cells (Set Timetable Enabled, Set School Open,
 * Reboot Requested, etc.) silently reject raw boolean writes — they require
 * the literal string "TRUE" or "FALSE" to match the checkbox's configured
 * checked/unchecked values. Without this coercion, string fields (hours,
 * names) would write fine while booleans would silently fail, which is
 * exactly the bug we saw: time hours updating but the timetable toggle,
 * Close School, and Reboot All doing nothing at the sheet level.
 */
function adminUpdateDevice(data) {
  console.log('[adminUpdateDevice] called with:', JSON.stringify(data));

  const deviceId = String(data.device_id || '').trim().toUpperCase();
  if (!deviceId) return { ok: false, error: 'missing_device_id' };

  // v45.1: Public lock around the read-find-write cycle.
  const lock = LockService.getPublicLock();
  try {
    lock.waitLock(30000);
  } catch (e) {
    return { ok: false, error: 'lock_timeout', detail: 'Could not acquire write lock (busy backend)' };
  }

  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const regSheet = ss.getSheetByName(REGISTRY_SHEET);
    if (!regSheet) return { ok: false, error: 'no_registry' };

    const lastRow = regSheet.getLastRow();
    if (lastRow < 2) return { ok: false, error: 'device_not_found', detail: deviceId };

    const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
    const col = {};
    headers.forEach((h, i) => { col[String(h).trim()] = i; });

    // Find device row
    const idCol = col['Device ID'];
    if (idCol === undefined) return { ok: false, error: 'registry_corrupt' };
    const ids = regSheet.getRange(2, idCol + 1, lastRow - 1, 1).getValues().flat();
    let rowIdx = -1;
    for (let i = 0; i < ids.length; i++) {
      if (String(ids[i]).trim().toUpperCase() === deviceId) { rowIdx = i + 2; break; }
    }
    if (rowIdx < 0) return { ok: false, error: 'device_not_found', detail: deviceId };

    // Map of snake_case field names to registry column headers
    const fieldMap = {
      'school_name':     'School Name',
      'org_type':        'Org Type',
      'org_name':        'Org Name',
      'org_urn':         'Org URN',
      'school_logo':     'School Logo',
      'teacher_name':    'Teacher Name',
      'holly_name':      'Holly Name',
      'school_address':  'School Address',
      'school_town':     'School Town',
      'school_county':   'School County',
      'school_region':   'School Region',
      'school_country':  'School Country',
      'school_postcode': 'School Postcode',
      'set_year_group':  'Set Year Group',
      'set_classroom':   'Set Classroom',
      'set_green_max':   'Set Green Max',
      'set_amber_max':   'Set Amber Max',
      'set_red_warn':    'Set Red Warn',
      'set_brightness':  'Set Brightness',
      'set_timetable_enabled': 'Set Timetable Enabled',
      'set_school_open': 'Set School Open',
      'set_reopen_date': 'Set Reopen Date',
      'set_send_count':  'Set SEND Count',
      'manual_override': 'Manual Override',
      'set_day_start':   'Set Day Start',
      'set_break_start': 'Set Break Start',
      'set_break_end':   'Set Break End',
      'set_lunch_start': 'Set Lunch Start',
      'set_lunch_end':   'Set Lunch End',
      'set_day_end':     'Set Day End',
      'set_pmbreak_start': 'Set Afternoon Break Start',
      'set_pmbreak_end':   'Set Afternoon Break End',
      'status':          'Status',
      'notes':           'Notes',
      'reboot_requested':'Reboot Requested',
      'room_test_requested':'Room Test Requested',
      'locate_requested':'Locate Requested',
      'set_self_test':   'Set Self Test',
      'set_room_cal_enabled':'Set Room Cal Enabled',
      'cancel_learning':'Cancel Learning',
      'sensitivity_profile':'Sensitivity Profile',
      // v49: Allow admin to clear the Last Error column. Pairs with the
      // stickiness suppression below — when admin clears, the cleared
      // string is remembered for 24 hours so Holly's firmware buffer
      // doesn't immediately repopulate the column on next telemetry.
      'last_error':      'Last Error',
      // v49: Allow admin to fire a one-shot "clear lastError buffer"
      // command to Holly's firmware (v5078+). Mirrors the existing
      // remote_action one-shot pattern. Auto-cleared after delivery.
      'clear_last_error':'Clear Last Error'
    };

    let updated = 0;
    let bumpConfig = false;

    // v49 (Option C): When admin clears last_error, remember the previously
    // stored string so we can suppress it from re-appearing if Holly's
    // firmware buffer (which we can't directly reach until B propagates)
    // sends it back on the next telemetry POST. Stored in script properties
    // with a 24-hour TTL — long enough for Option B firmware to arrive,
    // short enough that genuinely new identical errors will eventually surface.
    if (data.last_error === '' || data.last_error === null) {
      const errCol = col['Last Error'];
      if (errCol !== undefined) {
        const previousErr = String(regSheet.getRange(rowIdx, errCol + 1).getValue() || '').trim();
        if (previousErr.length > 0) {
          const props = PropertiesService.getScriptProperties();
          const stickKey = 'cleared_err_' + deviceId;
          props.setProperty(stickKey, JSON.stringify({
            err: previousErr,
            ts: Date.now()
          }));
          console.log('[adminUpdateDevice] Recorded cleared error for stickiness:', deviceId, previousErr);
        }
      }
    }

    Object.keys(data).forEach(key => {
      if (key === 'admin_action' || key === 'admin_key' || key === 'device_id') return;
      const colName = fieldMap[key];
      if (!colName) return;
      const c = col[colName];
      if (c === undefined) return;

      // v36: Coerce booleans to "TRUE"/"FALSE" strings. Checkbox-validated
      // cells silently reject JS boolean writes but accept the string forms.
      let writeVal = data[key];
      if (typeof writeVal === 'boolean') writeVal = writeVal ? 'TRUE' : 'FALSE';

      regSheet.getRange(rowIdx, c + 1).setValue(writeVal);
      updated++;
      if (CONFIG_COLUMNS.includes(colName)) bumpConfig = true;
    });

    // Bump config version if any config column was changed
    if (bumpConfig) {
      const cvCol = col['Config Version'];
      if (cvCol !== undefined) {
        const cv = Number(regSheet.getRange(rowIdx, cvCol + 1).getValue()) || 0;
        regSheet.getRange(rowIdx, cvCol + 1).setValue(cv + 1);
      }
    }

    SpreadsheetApp.flush();  // v45.1: force commit before lock release

    console.log('[adminUpdateDevice] wrote ' + updated + ' fields for ' + deviceId + ', bumpConfig=' + bumpConfig);
    return { ok: true, message: 'Updated ' + updated + ' fields for ' + deviceId, bumped_config: bumpConfig };
  } catch (err) {
    console.error('adminUpdateDevice error', err);
    return { ok: false, error: 'update_failed', detail: String(err) };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Delete a device from the registry (removes entire row).
 * Expects: device_id
 */
function adminDeleteDevice(data) {
  const deviceId = String(data.device_id || '').trim().toUpperCase();
  if (!deviceId) return { ok: false, error: 'missing_device_id' };

  // v45.1: Public lock around the read-find-delete cycle.
  const lock = LockService.getPublicLock();
  try {
    lock.waitLock(30000);
  } catch (e) {
    return { ok: false, error: 'lock_timeout', detail: 'Could not acquire write lock (busy backend)' };
  }

  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const regSheet = ss.getSheetByName(REGISTRY_SHEET);
    if (!regSheet) return { ok: false, error: 'no_registry' };

    const lastRow = regSheet.getLastRow();
    if (lastRow < 2) return { ok: false, error: 'device_not_found' };

    const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
    const col = {};
    headers.forEach(function(h, i) { col[String(h).trim()] = i; });

    const allData = regSheet.getRange(2, 1, lastRow - 1, headers.length).getValues();
    let deleteRowIdx = -1;
    let schoolName = '';

    for (let i = 0; i < allData.length; i++) {
      if (String(allData[i][col['Device ID']] || '').trim().toUpperCase() === deviceId) {
        deleteRowIdx = i;
        schoolName = String(allData[i][col['School Name']] || '').trim();
        break;
      }
    }

    if (deleteRowIdx < 0) return { ok: false, error: 'device_not_found', detail: deviceId };

    // Delete the row
    regSheet.deleteRow(deleteRowIdx + 2);

    // Check if any other devices remain at this school
    if (schoolName) {
      const remainingRows = regSheet.getLastRow();
      let schoolStillExists = false;
      if (remainingRows >= 2) {
        const remaining = regSheet.getRange(2, col['School Name'] + 1, remainingRows - 1, 1).getValues().flat();
        schoolStillExists = remaining.some(function(s) { return String(s).trim() === schoolName; });
      }

      // If no devices left at this school, archive the data tab
      if (!schoolStillExists) {
        const sheetName = sanitizeSheetName(schoolName);
        const dataTab = ss.getSheetByName(sheetName);
        if (dataTab) {
          const archiveName = '[ARCHIVED] ' + sheetName;
          // Check if archived name already exists (avoid duplicates)
          let finalName = archiveName;
          let counter = 1;
          while (ss.getSheetByName(finalName)) {
            finalName = archiveName + ' (' + counter + ')';
            counter++;
          }
          dataTab.setName(finalName);
          console.log('Archived data tab: ' + sheetName + ' → ' + finalName);
        }
      }
    }

    SpreadsheetApp.flush();  // v45.1: force commit before lock release

    return { ok: true, message: 'Device ' + deviceId + ' deleted' + (schoolName ? '' : '') };
  } catch (err) {
    console.error('adminDeleteDevice error', err);
    return { ok: false, error: 'delete_failed', detail: String(err) };
  } finally {
    lock.releaseLock();
  }
}

/* ==================== DASHBOARD DATA AGGREGATION ==================== */

/**
 * Builds the full classroom data model for the teacher dashboard.
 * Reads school data tab + registry, aggregates into per-classroom JSON.
 * Called via: ?action=dashboard_data&school=School+Name
 */
function buildDashboardData(schoolName) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = sanitizeSheetName(schoolName);
  const sheet = ss.getSheetByName(sheetName);
  
  // Get registry devices for this school
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: true, school: schoolName, classrooms: [] };
  
  const regLastRow = regSheet.getLastRow();
  const regLastCol = regSheet.getLastColumn();
  if (regLastRow < 2 || regLastCol < 1) return { ok: true, school: schoolName, classrooms: [] };
  
  const regHeaders = regSheet.getRange(1, 1, 1, regLastCol).getValues()[0].map(function(h) { return String(h).trim(); });
  const regData = regSheet.getRange(2, 1, regLastRow - 1, regLastCol).getValues();
  const rc = {};
  regHeaders.forEach(function(h, i) { rc[h] = i; });
  
  // Find all devices at this school
  var schoolDevices = [];
  for (var i = 0; i < regData.length; i++) {
    var devSchool = String(regData[i][rc['School Name']] || '').trim();
    if (devSchool === schoolName) {
      var dev = {};
      regHeaders.forEach(function(h, j) {
        dev[h] = regData[i][j];
      });
      schoolDevices.push(dev);
    }
  }
  
  if (schoolDevices.length === 0) return { ok: true, school: schoolName, classrooms: [] };

  // v51: Read pre-aggregated summary rows instead of raw telemetry.
  // Summary tab is maintained by refreshDashboardCacheTrigger every 5 mins.
  // 14 devices × ~30 days = ~420 rows max, queried in milliseconds.
  const summaryRows = _readSummaryForSchool_(schoolName);

  // Group summaries by device ID for quick per-device lookup
  const summaryByDevice = {};
  summaryRows.forEach(function(s) {
    const did = String(s['Device ID'] || '').trim().toUpperCase();
    if (!did) return;
    if (!summaryByDevice[did]) summaryByDevice[did] = [];
    summaryByDevice[did].push(s);
  });

  // Compute date windows we need (today, last 6 weeks)
  const tz = Session.getScriptTimeZone();
  const now = new Date();
  const todayKey = Utilities.formatDate(now, tz, 'yyyy-MM-dd');

  // Pre-compute the 6 weekly windows (each: { startKey, endKey } in YYYY-MM-DD)
  const weekWindows = [];
  for (var w = 5; w >= 0; w--) {
    const weekStart = new Date(now);
    weekStart.setDate(weekStart.getDate() - (w * 7 + weekStart.getDay()));
    weekStart.setHours(0, 0, 0, 0);
    const weekEnd = new Date(weekStart);
    weekEnd.setDate(weekEnd.getDate() + 7);
    weekWindows.push({
      startKey: Utilities.formatDate(weekStart, tz, 'yyyy-MM-dd'),
      endKey: Utilities.formatDate(weekEnd, tz, 'yyyy-MM-dd')
    });
  }

  // This Monday (for daily rewards Mon-Sat of current week)
  const thisMonday = new Date(now);
  thisMonday.setDate(thisMonday.getDate() - ((thisMonday.getDay() + 6) % 7));
  thisMonday.setHours(0, 0, 0, 0);
  const thisMondayKey = Utilities.formatDate(thisMonday, tz, 'yyyy-MM-dd');

  // Build classrooms by walking the registry devices and looking up summaries
  var classrooms = schoolDevices.map(function(dev) {
    const did = String(dev['Device ID'] || '').trim().toUpperCase();
    const devSummaries = summaryByDevice[did] || [];

    // --- Identity from registry (unchanged from v50) ---
    var classroom = String(dev['Active Classroom'] || dev['Set Classroom'] || '').trim();
    var teacher = String(dev['Teacher Name'] || '').trim();
    var hollyName = String(dev['Holly Name'] || '').trim();
    var firmware = String(dev['Firmware Version'] || '').trim();
    var lastSeen = dev['Last Seen'];
    var rssiVal = dev['WiFi RSSI'];
    var temp = Number(dev['Last Temperature']) || 0;
    var sendCount = Number(dev['Set SEND Count'] || dev['Active SEND Count'] || 0);
    var calStatus = String(dev['Cal Status'] || '').trim();
    var lastZone = String(dev['Last Zone'] || '').trim();
    var schoolOpen = String(dev['Active School Open'] || dev['Set School Open'] || '').toLowerCase();

    // --- Device status (6-state) ---
    var minsAgo = 999;
    if (lastSeen) {
      var lsDate = new Date(lastSeen);
      if (!isNaN(lsDate)) minsAgo = Math.floor((now.getTime() - lsDate.getTime()) / 60000);
    }
    var deviceStatus = 'offline';
    if (minsAgo <= 5) {
      if (calStatus === 'learning') deviceStatus = 'learning';
      else if (lastZone.indexOf('Holly Hop') >= 0) deviceStatus = 'hollyhop';
      else if (lastZone.indexOf('Sleeping') >= 0) deviceStatus = 'sleeping';
      else if (lastZone.indexOf('Dormant') >= 0 || schoolOpen === 'no' || schoolOpen === 'false') deviceStatus = 'dormant';
      else deviceStatus = 'online';
    }

    // --- Learning progress from registry (still pulled from registry, not summary) ---
    var learningPct = null;
    var lpRaw = String(dev['Learning Progress'] || '').trim();
    if (lpRaw && lpRaw.indexOf('%') >= 0) learningPct = parseInt(lpRaw);

    // --- Last Seen friendly string ---
    var lastSeenStr = '';
    if (minsAgo <= 1) lastSeenStr = '1 min ago';
    else if (minsAgo < 60) lastSeenStr = minsAgo + ' min ago';
    else if (minsAgo < 1440) lastSeenStr = Math.floor(minsAgo / 60) + ' hr ago';
    else lastSeenStr = Math.floor(minsAgo / 1440) + ' days ago';

    // --- Today's metrics (from summary row matching today's date) ---
    var todaySum = null;
    for (var i = 0; i < devSummaries.length; i++) {
      if (String(devSummaries[i]['Date'] || '').trim() === todayKey) {
        todaySum = devSummaries[i];
        break;
      }
    }
    var todayLessons = todaySum ? Number(todaySum['Lesson Rows'] || 0) : 0;
    var todayGreen = todaySum ? Number(todaySum['Green Rows'] || 0) : 0;
    var greenPct = todayLessons > 0 ? Math.round(todayGreen / todayLessons * 100) : 0;
    var rewardsToday = todaySum ? Number(todaySum['Reward Count'] || 0) : 0;
    var hopsToday = todaySum ? Number(todaySum['Hop Count'] || 0) : 0;

    // --- Total rewards (all time) ---
    var totalRw = 0;
    devSummaries.forEach(function(s) { totalRw += Number(s['Reward Count'] || 0); });

    // --- Today's score (matches old formula) ---
    var score = 0;
    if (todayLessons > 0) {
      var rewardRate = totalRw / Math.max(todayLessons / 60, 1);
      score = Math.round((greenPct / 100 * 8 + Math.min(rewardRate, 1) * 2) * 10) / 10;
      score = Math.max(0, Math.min(10, score));
    }

    // --- Weekly scores (last 6 weeks) — aggregate across summary rows in each window ---
    var weekScores = [];
    weekWindows.forEach(function(wnd) {
      var wLessons = 0, wGreen = 0, wRewards = 0;
      devSummaries.forEach(function(s) {
        var dKey = String(s['Date'] || '').trim();
        if (dKey >= wnd.startKey && dKey < wnd.endKey) {
          wLessons += Number(s['Lesson Rows'] || 0);
          wGreen   += Number(s['Green Rows']  || 0);
          wRewards += Number(s['Reward Count']|| 0);
        }
      });
      if (wLessons > 0) {
        var wGreenPct = wGreen / wLessons * 100;
        var wRewardRate = wRewards / Math.max(wLessons / 60, 1);
        var wScore = Math.round((wGreenPct / 100 * 8 + Math.min(wRewardRate, 1) * 2) * 10) / 10;
        weekScores.push(Math.max(0, Math.min(10, wScore)));
      } else {
        // Fall back to previous week's score if no data
        weekScores.push(weekScores.length > 0 ? weekScores[weekScores.length - 1] : score);
      }
    });

    // --- Daily rewards Mon-Sat of current week ---
    var dailyRw = [0, 0, 0, 0, 0, 0];
    devSummaries.forEach(function(s) {
      var dKey = String(s['Date'] || '').trim();
      if (dKey >= thisMondayKey) {
        var d = new Date(dKey + 'T00:00:00');
        if (!isNaN(d)) {
          var dayIdx = (d.getDay() + 6) % 7; // 0=Mon
          if (dayIdx < 6) dailyRw[dayIdx] += Number(s['Reward Count'] || 0);
        }
      }
    });

    // --- SEND data: spikes and transition dB per week ---
    var weeklySpikes = [];
    var weeklyTransDb = [];
    weekWindows.forEach(function(wnd) {
      var spikeSum = 0, spikeCount = 0, transSum = 0, transCount = 0;
      devSummaries.forEach(function(s) {
        var dKey = String(s['Date'] || '').trim();
        if (dKey >= wnd.startKey && dKey < wnd.endKey) {
          spikeSum += Number(s['Spike Sum'] || 0);
          spikeCount += Number(s['Spike Count'] || 0);
          transSum += Number(s['Trans Sum'] || 0);
          transCount += Number(s['Trans Count'] || 0);
        }
      });
      weeklySpikes.push(spikeCount > 0 ? Math.round(spikeSum / spikeCount * 10) / 10 : 0);
      weeklyTransDb.push(transCount > 0 ? Math.round(transSum / transCount * 10) / 10 : 0);
    });

    // --- Year group from registry ---
    var yearGroup = String(dev['Set Year Group'] || dev['Active Year Group'] || '').trim();
    var yearBand = String(dev['Active Year Band'] || '').trim();

    return {
      name: classroom || 'Device ' + did,
      teacher: teacher,
      score: score,
      temp: Math.round(temp * 10) / 10,
      rewards: rewardsToday,
      greenPct: greenPct,
      weekly: weekScores,
      dailyRw: dailyRw,
      totalRw: totalRw,
      hollyName: hollyName,
      deviceId: did,
      firmware: firmware,
      lastSeen: lastSeenStr,
      rssi: rssiVal !== '' && rssiVal !== null ? Number(rssiVal) : null,
      deviceStatus: deviceStatus,
      learningPct: learningPct,
      hops: hopsToday,
      sendCount: sendCount,
      sensitivityProfile: String(dev['Sensitivity Profile'] || 'standard').toLowerCase().trim(),
      voicePercent: Math.round(Number(dev['Voice %'] || 0)),
      spikes: weeklySpikes,
      transDb: weeklyTransDb,
      yearGroup: yearGroup,
      yearBand: yearBand,
      micFloor: Number(dev['Mic Floor']) || null,
      micFloorValid: String(dev['Mic Floor Valid'] || '').toLowerCase() === 'yes',
      micFloorDate: String(dev['Mic Floor Date'] || '').trim(),
      locateRequested: String(dev['Locate Requested'] || '').trim(),
      calStatus: calStatus,
      calStartDate: dev['Cal Start Date'] ? Utilities.formatDate(new Date(dev['Cal Start Date']), tz, 'yyyy-MM-dd HH:mm:ss') : '',
      // v53: live device_phase + window state from registry (firmware v5083+).
      // Empty string for pre-5083 devices that don't emit these fields.
      device_phase: String(dev['Active Device Phase'] || '').trim(),
      ws: String(dev['Active Window State'] || '').trim()
    };
  });

  classrooms.sort(function(a, b) { return b.score - a.score; });

  return {
    ok: true,
    school: schoolName,
    classrooms: classrooms,
    generated: new Date().toISOString(),
    dataRows: summaryRows.length,
    fromCache: 'summary_tab'
  };
}

/* ==================== DAILY DATA (Pick a date report) ==================== */

/**
 * Builds real day-level data for the Daily Report page.
 * Returns per-classroom metrics for a single calendar date:
 *   avgDb, greenPct/amberPct/redPct, rewards, hops, dataPoints, score.
 * Plus school-wide aggregates so the dashboard doesn't have to re-loop.
 *
 * Called via: ?action=daily_data&school=School+Name&date=YYYY-MM-DD
 */
function buildDailyData(schoolName, dateStr) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = sanitizeSheetName(schoolName);
  const sheet = ss.getSheetByName(sheetName);

  // Parse target date — build day window in script timezone so comparisons line up
  // with how rows were written via Utilities.formatDate elsewhere.
  const tz = Session.getScriptTimeZone();

  // Registry lookup for device identity
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) {
    return { ok: true, school: schoolName, date: dateStr, classrooms: [], schoolwide: emptyDailySchoolwide_(), dataPoints: 0 };
  }
  const regLastRow = regSheet.getLastRow();
  const regLastCol = regSheet.getLastColumn();
  if (regLastRow < 2 || regLastCol < 1) {
    return { ok: true, school: schoolName, date: dateStr, classrooms: [], schoolwide: emptyDailySchoolwide_(), dataPoints: 0 };
  }

  const regHeaders = regSheet.getRange(1, 1, 1, regLastCol).getValues()[0].map(function(h) { return String(h).trim(); });
  const regData = regSheet.getRange(2, 1, regLastRow - 1, regLastCol).getValues();
  const rc = {};
  regHeaders.forEach(function(h, i) { rc[h] = i; });

  var schoolDevices = [];
  for (var i = 0; i < regData.length; i++) {
    var devSchool = String(regData[i][rc['School Name']] || '').trim();
    if (devSchool === schoolName) {
      var dev = {};
      regHeaders.forEach(function(h, j) { dev[h] = regData[i][j]; });
      schoolDevices.push(dev);
    }
  }
  if (schoolDevices.length === 0) {
    return { ok: true, school: schoolName, date: dateStr, classrooms: [], schoolwide: emptyDailySchoolwide_(), dataPoints: 0 };
  }

  // v51: Read school tab BUT filter to target date during the read,
  // not after. On big tabs (10K+ rows) building object dictionaries for
  // every row only to throw 99% away is wasteful. Cheap pre-filter on
  // the Date column halves response time on large tabs.
  var allRows = [];
  if (sheet) {
    var lastRow = sheet.getLastRow();
    var lastCol = sheet.getLastColumn();
    if (lastRow >= 2 && lastCol >= 1) {
      var dataHeaders = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function(h) { return String(h).trim(); });
      var rawData = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();
      var dateColIdx = dataHeaders.indexOf('Date');
      var dtColIdx = dataHeaders.indexOf('DateTime');
      rawData.forEach(function(row) {
        // Pre-filter: skip rows that aren't from target date
        var rowDate = '';
        if (dateColIdx >= 0) {
          var dv = row[dateColIdx];
          if (dv instanceof Date) rowDate = Utilities.formatDate(dv, tz, 'yyyy-MM-dd');
          else if (dv) {
            var ds = String(dv).trim();
            if (/^\d{4}-\d{2}-\d{2}/.test(ds)) rowDate = ds.substring(0, 10);
          }
        }
        if (rowDate === '' && dtColIdx >= 0) {
          var dt = row[dtColIdx];
          if (dt instanceof Date) rowDate = Utilities.formatDate(dt, tz, 'yyyy-MM-dd');
        }
        if (rowDate !== dateStr) return; // skip — not our date
        var obj = {};
        dataHeaders.forEach(function(h, j) { obj[h] = row[j]; });
        allRows.push(obj);
      });
    }
  }

  // No further date filter needed — pre-filtered above
  function rowDateStr(r) { return dateStr; }
  var dayRows = allRows;

  // Group by Device ID
  var byDevice = {};
  dayRows.forEach(function(r) {
    var did = String(r['Device ID'] || '').trim().toUpperCase();
    if (!did) return;
    if (!byDevice[did]) byDevice[did] = [];
    byDevice[did].push(r);
  });

  // Build per-classroom day summaries
  var classroomsOut = schoolDevices.map(function(dev) {
    var did = String(dev['Device ID'] || '').trim().toUpperCase();
    var devRows = byDevice[did] || [];

    var classroom = String(dev['Active Classroom'] || dev['Set Classroom'] || '').trim();
    var teacher = String(dev['Teacher Name'] || '').trim();
    var hollyName = String(dev['Holly Name'] || '').trim();
    var yearGroup = String(dev['Set Year Group'] || dev['Active Year Group'] || '').trim();

    // Lesson rows: dB present, not events
    var lessonRows = devRows.filter(function(r) {
      return !r['Event Type'] && r['Sound dB'] !== '' && r['Sound dB'] !== null && !isNaN(Number(r['Sound dB']));
    });

    var dataPoints = lessonRows.length;

    // Avg dB
    var dbSum = 0;
    lessonRows.forEach(function(r) { dbSum += Number(r['Sound dB']) || 0; });
    var avgDb = dataPoints > 0 ? Math.round(dbSum / dataPoints * 10) / 10 : null;

    // LED zone distribution — ignore Holly Hop, Sleeping, Dormant for clean % split
    var green = 0, amber = 0, red = 0, reward = 0;
    lessonRows.forEach(function(r) {
      var st = String(r['LED State'] || '').trim();
      if (st === 'Green') green++;
      else if (st === 'Reward (Quiet)') { green++; reward++; }
      else if (st === 'Amber') amber++;
      else if (st === 'Red' || st === 'Red Warn') red++;
    });
    var zoneTotal = green + amber + red;
    var greenPct = zoneTotal > 0 ? Math.round(green / zoneTotal * 100) : 0;
    var amberPct = zoneTotal > 0 ? Math.round(amber / zoneTotal * 100) : 0;
    var redPct   = zoneTotal > 0 ? 100 - greenPct - amberPct : 0; // ensures sum = 100

    // Rewards: Quiet Reward = yes
    var rewards = lessonRows.filter(function(r) {
      return String(r['Quiet Reward'] || '').trim().toLowerCase() === 'yes';
    }).length;

    // Hops: Event Type = HOLLY_HOP, Event = START (on this day)
    var hops = devRows.filter(function(r) {
      var evType = String(r['Event Type'] || '').trim();
      var ev = String(r['Event'] || '').trim();
      return evType === 'HOLLY_HOP' && ev === 'START';
    }).length;

    // Score — same formula as buildDashboardData for consistency
    var score = 0;
    if (dataPoints > 0) {
      var rewardRate = rewards / Math.max(dataPoints / 60, 1);
      score = Math.round((greenPct / 100 * 8 + Math.min(rewardRate, 1) * 2) * 10) / 10;
      score = Math.max(0, Math.min(10, score));
    }

    // First/last activity time (useful context for the dashboard)
    var firstTime = null, lastTime = null;
    lessonRows.forEach(function(r) {
      var t = r['DateTime'] || r['Date'];
      if (!(t instanceof Date)) t = new Date(t);
      if (isNaN(t.getTime())) return;
      if (!firstTime || t < firstTime) firstTime = t;
      if (!lastTime || t > lastTime) lastTime = t;
    });

    // --- Time series: dB + zone per reading, bucketed to 5-min intervals ---
    // Teachers don't need per-second detail; 5-min buckets keep charts readable
    // and avoid exposing thousands of points.
    var buckets = {}; // key: "HH:MM" (aligned to 5 min) -> {dbSum, dbCount, zoneCounts}
    lessonRows.forEach(function(r) {
      var t = r['DateTime'] || r['Date'];
      if (!(t instanceof Date)) t = new Date(t);
      if (isNaN(t.getTime())) return;
      var hh = t.getHours();
      var mm = Math.floor(t.getMinutes() / 5) * 5;
      var key = String(hh).padStart(2,'0') + ':' + String(mm).padStart(2,'0');
      if (!buckets[key]) buckets[key] = { dbSum: 0, dbCount: 0, zones: { Green:0, Amber:0, Red:0 } };
      var dbVal = Number(r['Sound dB']);
      if (!isNaN(dbVal)) { buckets[key].dbSum += dbVal; buckets[key].dbCount += 1; }
      var st = String(r['LED State'] || '').trim();
      if (st === 'Green' || st === 'Reward (Quiet)') buckets[key].zones.Green += 1;
      else if (st === 'Amber') buckets[key].zones.Amber += 1;
      else if (st === 'Red' || st === 'Red Warn') buckets[key].zones.Red += 1;
    });
    var timeSeries = Object.keys(buckets).sort().map(function(k) {
      var b = buckets[k];
      var avgBucketDb = b.dbCount > 0 ? Math.round(b.dbSum / b.dbCount * 10) / 10 : null;
      // Dominant zone in this bucket
      var topZone = 'Green', topCount = b.zones.Green;
      if (b.zones.Amber > topCount) { topZone = 'Amber'; topCount = b.zones.Amber; }
      if (b.zones.Red > topCount)   { topZone = 'Red';   topCount = b.zones.Red; }
      return { t: k, db: avgBucketDb, zone: topZone };
    });

    // --- Events timeline: hops and rewards with timestamps ---
    var events = [];
    devRows.forEach(function(r) {
      var t = r['DateTime'] || r['Date'];
      if (!(t instanceof Date)) t = new Date(t);
      if (isNaN(t.getTime())) return;
      var timeStr = Utilities.formatDate(t, tz, 'HH:mm');
      var evType = String(r['Event Type'] || '').trim();
      var ev = String(r['Event'] || '').trim();
      if (evType === 'HOLLY_HOP') {
        events.push({ t: timeStr, kind: 'hop_' + (ev === 'START' ? 'start' : 'end'), label: 'Holly Hop ' + (ev === 'START' ? 'started' : 'ended') });
      } else if (String(r['Quiet Reward'] || '').trim().toLowerCase() === 'yes') {
        events.push({ t: timeStr, kind: 'reward', label: 'Quiet Reward earned' });
      }
    });
    // De-duplicate rewards in the same minute (multiple rows logging the same moment)
    var seenReward = {};
    events = events.filter(function(e) {
      if (e.kind !== 'reward') return true;
      if (seenReward[e.t]) return false;
      seenReward[e.t] = true;
      return true;
    });
    events.sort(function(a,b) { return a.t < b.t ? -1 : a.t > b.t ? 1 : 0; });

    return {
      name: classroom || 'Device ' + did,
      teacher: teacher,
      hollyName: hollyName,
      deviceId: did,
      yearGroup: yearGroup,
      dataPoints: dataPoints,
      avgDb: avgDb,
      greenPct: greenPct,
      amberPct: amberPct,
      redPct: redPct,
      rewards: rewards,
      hops: hops,
      score: score,
      firstReading: firstTime ? Utilities.formatDate(firstTime, tz, 'HH:mm') : null,
      lastReading: lastTime ? Utilities.formatDate(lastTime, tz, 'HH:mm') : null,
      timeSeries: timeSeries,
      events: events,
      // v53: live device_phase + window state from registry (firmware v5083+).
      // Empty string for pre-5083 devices that don't emit these fields.
      device_phase: String(dev['Active Device Phase'] || '').trim(),
      ws: String(dev['Active Window State'] || '').trim()
    };
  });

  // School-wide aggregates
  var active = classroomsOut.filter(function(c) { return c.dataPoints > 0; });
  var schoolwide = emptyDailySchoolwide_();
  schoolwide.classroomsReporting = active.length;
  schoolwide.classroomsTotal = classroomsOut.length;
  if (active.length > 0) {
    var sumDb = 0, sumRw = 0, sumHops = 0, sumGreen = 0, sumAmber = 0, sumRed = 0;
    var bestScore = -1, worstScore = 11, bestName = '', worstName = '';
    active.forEach(function(c) {
      sumDb    += c.avgDb || 0;
      sumRw    += c.rewards;
      sumHops  += c.hops;
      sumGreen += c.greenPct;
      sumAmber += c.amberPct;
      sumRed   += c.redPct;
      if (c.score > bestScore)  { bestScore = c.score; bestName = c.name; }
      if (c.score < worstScore) { worstScore = c.score; worstName = c.name; }
    });
    schoolwide.avgDb       = Math.round(sumDb / active.length * 10) / 10;
    schoolwide.totalRewards = sumRw;
    schoolwide.totalHops    = sumHops;
    schoolwide.avgGreenPct  = Math.round(sumGreen / active.length);
    schoolwide.avgAmberPct  = Math.round(sumAmber / active.length);
    schoolwide.avgRedPct    = Math.round(sumRed / active.length);
    schoolwide.calmestName  = bestName;
    schoolwide.calmestScore = bestScore;
    schoolwide.busiestName  = worstName;
    schoolwide.busiestScore = worstScore;

    // School-wide time series: average dB across all classrooms per 5-min bucket
    var schoolBuckets = {};
    active.forEach(function(c) {
      (c.timeSeries || []).forEach(function(pt) {
        if (pt.db === null || pt.db === undefined) return;
        if (!schoolBuckets[pt.t]) schoolBuckets[pt.t] = { sum: 0, count: 0 };
        schoolBuckets[pt.t].sum += pt.db;
        schoolBuckets[pt.t].count += 1;
      });
    });
    schoolwide.timeSeries = Object.keys(schoolBuckets).sort().map(function(k) {
      var b = schoolBuckets[k];
      return { t: k, db: Math.round(b.sum / b.count * 10) / 10 };
    });
  }

  // Sort classrooms by score descending
  classroomsOut.sort(function(a, b) { return b.score - a.score; });

  return {
    ok: true,
    school: schoolName,
    date: dateStr,
    classrooms: classroomsOut,
    schoolwide: schoolwide,
    dataPoints: dayRows.length,
    generated: new Date().toISOString()
  };
}

  return {
    classroomsReporting: 0,
    classroomsTotal: 0,
    avgDb: null,
    totalRewards: 0,
    totalHops: 0,
    avgGreenPct: 0,
    avgAmberPct: 0,
    avgRedPct: 0,
    calmestName: null,
    calmestScore: null,
    busiestName: null,
    busiestScore: null
  };
}

/**
 * Returns school identity data for the teacher dashboard About tab.
 * Looks up ANY device at the school to get school-level fields.
 * Called via: ?action=about_school&device_id=EF0C14
 * No admin key required.
 */
function buildAboutSchool(deviceId) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: false, error: 'no_registry' };

  const lastRow = regSheet.getLastRow();
  if (lastRow < 2) return { ok: false, error: 'empty_registry' };

  const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0]
    .map(h => String(h).trim());
  const data = regSheet.getRange(2, 1, lastRow - 1, headers.length).getValues();
  const col = {};
  headers.forEach((h, i) => { col[h] = i; });

  const g = (h, row) => String(row[col[h]] || '').trim();

  // Find the device row
  let deviceRow = null;
  for (let i = 0; i < data.length; i++) {
    if (String(data[i][col['Device ID']] || '').trim().toUpperCase() === deviceId) {
      deviceRow = data[i];
      break;
    }
  }
  if (!deviceRow) return { ok: false, error: 'device_not_found', detail: deviceId };

  const schoolName = g('School Name', deviceRow);

  // Count all devices at this school
  const schoolDevices = data.filter(row =>
    String(row[col['School Name']] || '').trim() === schoolName
  );

  return {
    ok: true,
    schoolName:    schoolName,
    orgType:       g('Org Type',       deviceRow),
    orgName:       g('Org Name',       deviceRow),
    orgUrn:        g('Org URN',        deviceRow),
    schoolLogo:    g('School Logo',    deviceRow),
    schoolAddress: g('School Address', deviceRow),
    schoolTown:    g('School Town',    deviceRow),
    schoolCounty:  g('School County',  deviceRow),
    schoolRegion:  g('School Region',  deviceRow),
    schoolCountry: g('School Country', deviceRow),
    schoolPostcode:g('School Postcode',deviceRow),
    deviceCount:   schoolDevices.length
  };
}

/* ==================== HELPERS ==================== */

function numOrEmpty(v) {
  if (v === '' || v == null || v === undefined) return '';
  const n = Number(v);
  return isNaN(n) ? '' : n;
}

function strOrEmpty(v) {
  if (v === '' || v == null || v === undefined) return '';
  return String(v).trim();
}

function boolOrEmpty(v) {
  if (v === '' || v == null || v === undefined) return '';
  if (typeof v === 'boolean') return v;
  const s = String(v).toLowerCase().trim();
  if (['true','1','yes'].includes(s)) return true;
  if (['false','0','no'].includes(s)) return false;
  return '';
}

function sanitizeSheetName(name) {
  let s = (name || '').trim();
  if (!s) return 'Unknown School';
  s = s.replace(/[:\\\/\?\*\[\]]/g, ' ').replace(/\s+/g, ' ').trim();
  if (s.length > 90) s = s.slice(0, 90);
  return s || 'Unknown School';
}

function ensureSheetAndHeaders(sheetName) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) sheet = ss.insertSheet(sheetName);

  const lastCol = Math.max(sheet.getLastColumn(), TARGET_HEADERS.length);
  const current = lastCol > 0 ? sheet.getRange(1,1,1,lastCol).getValues()[0] : [];
  const existing = current.filter(v => String(v).trim() !== '');

  // For a brand new or wiped sheet, use TARGET_HEADERS exactly
  let headers;
  if (existing.length === 0) {
    headers = TARGET_HEADERS.slice();
  } else {
    // Existing sheet — only add missing headers, don't remove old ones
    headers = existing.slice();
    TARGET_HEADERS.forEach(h => { if (!headers.includes(h)) headers.push(h); });
  }

  const headerRow = lastCol > 0 ? sheet.getRange(1,1,1,Math.max(lastCol, headers.length)).getValues()[0] : [];
  const needsRewrite = headers.length !== headerRow.length || headers.some((h,i)=>h!==headerRow[i]);
  if (needsRewrite) {
    sheet.getRange(1,1,1,headers.length).setValues([headers]);
    sheet.getRange(1,1,1,headers.length).setFontWeight('bold');
  }

  // Formats
  const maxRows = Math.max(sheet.getMaxRows(), 1000);
  const col = (name) => headers.indexOf(name) + 1;
  const cDate = col('Date');
  const cTime = col('Time');
  const cDT   = col('DateTime');

  if (cDate > 0) sheet.getRange(2, cDate, maxRows-1, 1).setNumberFormat('yyyy-mm-dd');
  if (cTime > 0) sheet.getRange(2, cTime, maxRows-1, 1).setNumberFormat('HH:mm:ss');
  if (cDT   > 0) sheet.getRange(2, cDT,   maxRows-1, 1).setNumberFormat('yyyy-mm-dd HH:mm:ss');

  return { sheet, headers };
}

function parseIncoming(e) {
  let body = '', data = {}, parseError = null, contentType = '';
  try {
    if (e && e.postData) {
      contentType = e.postData.type || '';
      body = e.postData.contents || '';
      if (body && body.trim().startsWith('{')) {
        try { data = JSON.parse(body); }
        catch (err) { parseError = 'JSON parse failed: ' + String(err); data = {}; }
      } else {
        data = Object.assign({}, e.parameter || {});
        if (!body) body = JSON.stringify(data);
      }
    }
  } catch (err) {
    parseError = 'parseIncoming exception: ' + String(err);
  }
  return { data, raw: body, contentType, parseError };
}

function buildRow(headers, data) {
  const ts = toDateSafe(data.timestamp) || new Date();
  const dateOnly = new Date(ts.getFullYear(), ts.getMonth(), ts.getDate());
  const timeFraction = (ts.getHours()*3600 + ts.getMinutes()*60 + ts.getSeconds()) / 86400;

  const num = v => (v===''||v==null ? '' : (isNaN(Number(v)) ? v : Number(v)));
  const bool = v => {
    if (v===''||v==null) return '';
    if (typeof v==='boolean') return v;
    const s = String(v).toLowerCase();
    if (['true','1','yes','y'].includes(s)) return true;
    if (['false','0','no','n'].includes(s)) return false;
    return v;
  };

  const get = {
    'Date':             () => dateOnly,
    'Time':             () => timeFraction,
    'DateTime':         () => ts,
    'Device ID':        () => data.device_id || data['Device ID'] || '',
    'Holly Name':       () => data.holly_name || data['Holly Name'] || '',
    'Classroom':        () => data.classroom || data['Classroom'] || '',
    'Sound dB':         () => num(data.sound_dB ?? data['Sound dB']),
    'Voice %':          () => num(data.voice_percent ?? data['Voice %']),
    'LED State':        () => data.led_state || data['LED State'] || '',
    'Temperature (°C)': () => num(data.temperature_C ?? data['Temperature (°C)']),
    'Humidity (%)':     () => num(data.humidity_percent ?? data['Humidity (%)']),
    'Th Green Max':     () => num(data.th_green_max ?? data['Th Green Max']),
    'Th Amber Max':     () => num(data.th_amber_max ?? data['Th Amber Max']),
    'Th Red Warn':      () => num(data.th_red_warn  ?? data['Th Red Warn']),
    'Quiet Reward':     () => data.quiet_reward || data['Quiet Reward'] || '',
    'Holly Hop':        () => bool(data.holly_hop ?? data['Holly Hop']),
    'Learning Progress':() => data.learning_progress || data['Learning Progress'] || '',
    'Spike Count':      () => num(data.spike_count ?? data['Spike Count']),
    'Transition Avg dB':() => num(data.transition_avg_db ?? data['Transition Avg dB']),
    'SEND Count':       () => num(data.send_count ?? data['SEND Count']),
    'Event Type':       () => data.event_type || data['Event Type'] || '',
    'Event':            () => data.event || data['Event'] || '',
  };

  return headers.map(h => (get[h] ? get[h]() : ''));
}

function toDateSafe(v) {
  if (!v && v !== 0) return null;
  try {
    if (v instanceof Date) return v;
    if (typeof v === 'number') return new Date(v);
    if (typeof v === 'string') {
      const s = v.replace(' ', 'T');
      const d = new Date(s);
      if (!isNaN(d.getTime())) return d;
    }
  } catch (_) {}
  return null;
}

  return ContentService

    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Run once to create a clean Device Registry with v10 columns.
 * Safe to run multiple times — only adds missing columns and devices.
 *
 * IMPORTANT: If you want a totally clean registry, delete the
 * "Device Registry" tab first, then run this.
 */
function setupDeviceRegistry() {
  const devices = [
    // Devices are now registered via the admin dashboard API.
    // No hardcoded devices needed.
  ];

  const ss = SpreadsheetApp.openById(SHEET_ID);
  let regSheet = ss.getSheetByName(REGISTRY_SHEET);

  if (!regSheet) {
    regSheet = ss.insertSheet(REGISTRY_SHEET);
    regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setValues([REGISTRY_HEADERS]);
    regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setFontWeight('bold');
  } else {
    // Expand existing — add missing columns only
    const lastCol = regSheet.getLastColumn();
    if (lastCol === 0) {
      // Sheet exists but is completely empty — write full headers
      regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setValues([REGISTRY_HEADERS]);
      regSheet.getRange(1, 1, 1, REGISTRY_HEADERS.length).setFontWeight('bold');
    } else {
      const existingHeaders = regSheet.getRange(1, 1, 1, lastCol).getValues()[0]
        .map(h => String(h).trim());

      let added = 0;
      REGISTRY_HEADERS.forEach(h => {
        if (!existingHeaders.includes(h)) {
          const nextCol = existingHeaders.length + added + 1;
          regSheet.getRange(1, nextCol).setValue(h).setFontWeight('bold');
          added++;
        }
      });

      if (added > 0) console.log('Added ' + added + ' new columns to registry');
    }
  }

  // Add missing devices
  const lastRow = regSheet.getLastRow();
  const existingIds = new Set();
  if (lastRow >= 2) {
    const existing = regSheet.getRange(2, 1, lastRow - 1, 1).getValues().flat();
    existing.forEach(id => existingIds.add(String(id).trim().toUpperCase()));
  }

  const finalHeaders = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0]
    .map(h => String(h).trim());

  let devicesAdded = 0;
  devices.forEach(dev => {
    const id = dev.id.toUpperCase();
    if (!existingIds.has(id)) {
      const row = finalHeaders.map(h => {
        if (h === 'Device ID')      return id;
        if (h === 'School Name')    return dev.school;
        if (h === 'School Address') return dev.address || '';
        if (h === 'School Town')    return dev.town || '';
        if (h === 'School County')  return dev.county || '';
        if (h === 'School Region')  return dev.region || '';
        if (h === 'School Country') return dev.country || '';
        if (h === 'School Postcode')return dev.postcode || '';
        if (h === 'Config Version') return 0;
        if (h === 'Status')         return 'active';
        return '';
      });
      regSheet.appendRow(row);
      existingIds.add(id);
      devicesAdded++;
    }
  });

  Logger.log('Registry setup complete (v17). Columns: ' + finalHeaders.length +
    ', Devices added: ' + devicesAdded + ', Total devices: ' + existingIds.size);
}

/* ==================== CLEANUP HELPER ==================== */

/**
 * Run this once to wipe all data tabs and start fresh.
 * Deletes: Logs, Raw, and all school-named tabs.
 * Preserves: Device Registry (headers + device rows, clears health data).
 *
 * After running, new POSTs will create clean school tabs with v10 headers.
 */
function wipeAndStartFresh() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const preserve = new Set([REGISTRY_SHEET]);
  
  // Delete all non-registry tabs
  const sheets = ss.getSheets();
  sheets.forEach(sh => {
    const name = sh.getName();
    if (!preserve.has(name)) {
      // Can't delete the last sheet — so create a temp if needed
      if (sheets.length > 1 || ss.getSheets().length > 1) {
        try { ss.deleteSheet(sh); }
        catch (e) { console.error('Could not delete sheet: ' + name, e); }
      }
    }
  });

  // Clear device-reported columns in registry (keep identity + admin config)
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (regSheet) {
    const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
    const lastRow = regSheet.getLastRow();
    if (lastRow < 2) return;

    // Columns to clear (device-reported, not admin-set)
    const clearColumns = [
      'Confirmed Version', 'Active Green', 'Active Amber', 'Active Red',
      'Active Brightness', 'Active Year Group', 'Active Year Band',
      'Active Classroom', 'Active Timetable Enabled', 'Active School Open',
      'Active SEND Count',
      'Last Zone', 'Last Temperature',
      'Firmware Version', 'IP Address',
      'MAC Address', 'WiFi RSSI', 'Last Seen', 'Uptime Minutes',
      'Last Reset Reason', 'Last Error',
      'Active Day Start', 'Active Break Start', 'Active Break End',
      'Active Lunch Start', 'Active Lunch End', 'Active Day End',
      'Cal Status', 'Cal Mean dB', 'Cal SD', 'Cal Min dB', 'Cal Max dB',
      'Cal Samples', 'Cal Green', 'Cal Amber', 'Cal Red',
      'Cal Start Date', 'Cal Complete Date',

function doGet_insights(e) {
  const p = (e && e.parameter) || {};

  // Public endpoint — dashboard reads insights for a school
  if (p.action === 'insights') {
    const school = (p.school || '').trim();
    if (!school) return json_ins({ ok: false, error: 'missing_school' });
    const period = (p.period_type || '').trim(); // optional filter
    return json_ins(readInsightsForSchool(school, period));
  }

  // Admin — manually trigger a regeneration
  if (p.action === 'regenerate_insight' && p.admin_key === ADMIN_KEY) {
    const school = (p.school || '').trim();
    const period = (p.period_type || 'monthly').trim();
    return json_ins(generateInsightForSchool(school, period, true /*force*/));
  }

  // Admin — run full batch now
  if (p.action === 'run_insights_now' && p.admin_key === ADMIN_KEY) {
    return json_ins(generateAllInsights());
  }

  return null; // not an insights request — let main doGet handle
}

/* ==================== PERIOD HELPERS ==================== */

function getPeriodRange(periodType, refDate) {
  const d = refDate ? new Date(refDate) : new Date();
  let start, end, key;

  if (periodType === 'weekly') {
    // ISO week Monday–Sunday — use the PREVIOUS completed week
    const day = (d.getDay() + 6) % 7; // Mon=0
    start = new Date(d); start.setDate(d.getDate() - day - 7); start.setHours(0,0,0,0);
    end   = new Date(start); end.setDate(start.getDate() + 6); end.setHours(23,59,59,999);
    // ISO week number
    const tmp = new Date(start); tmp.setDate(tmp.getDate() + 3);
    const w1  = new Date(tmp.getFullYear(), 0, 4);
    const weekNum = 1 + Math.round(((tmp - w1) / 86400000 - 3 + (w1.getDay() + 6) % 7) / 7);
    key = start.getFullYear() + '-W' + String(weekNum).padStart(2, '0');
  }
  else if (periodType === 'monthly') {
    // Previous completed calendar month
    const y = d.getFullYear(), m = d.getMonth();
    start = new Date(y, m - 1, 1);
    end   = new Date(y, m, 0, 23, 59, 59, 999);
    key   = start.getFullYear() + '-' + String(start.getMonth() + 1).padStart(2, '0');
  }
  else if (periodType === 'halfterm') {
    // Approximate: last 6 weeks. Replace with your getCurrentTerm() if you have one.
    end = new Date(d); end.setHours(23,59,59,999);
    start = new Date(end); start.setDate(end.getDate() - 42);
    key = 'HT-' + Utilities.formatDate(end, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  else if (periodType === 'termly') {
    end = new Date(d); end.setHours(23,59,59,999);
    start = new Date(end); start.setDate(end.getDate() - 90);
    key = 'T-' + Utilities.formatDate(end, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  else if (periodType === 'yearly') {
    // Academic year Sept–Aug
    const m = d.getMonth();
    const startYear = m >= 8 ? d.getFullYear() : d.getFullYear() - 1;
    start = new Date(startYear, 8, 1); // 1 Sept
    end   = new Date(startYear + 1, 7, 31, 23, 59, 59, 999); // 31 Aug
    key = 'AY-' + startYear + '-' + (startYear + 1);
  }
  else {
    throw new Error('Unknown period type: ' + periodType);
  }

  return { start: start, end: end, key: key };
}


/* ==================== STATS COMPUTATION ==================== */

/**
 * Builds the structured stats object we hand to the LLM.
 * AI does NOT do arithmetic. This function does the arithmetic deterministically.
 */
function computePeriodStats(schoolName, periodType) {
  const range = getPeriodRange(periodType);
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = sanitizeSheetName(schoolName);
  const dataSheet = ss.getSheetByName(sheetName);
  const regSheet  = ss.getSheetByName(REGISTRY_SHEET);

  if (!regSheet) throw new Error('No registry');

  // Registry lookup for this school's devices
  const regLast = regSheet.getLastRow();
  const regHeaders = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0].map(String);
  const regCol = {}; regHeaders.forEach((h, i) => regCol[h.trim()] = i);
  const regRows = regLast >= 2 ? regSheet.getRange(2, 1, regLast - 1, regHeaders.length).getValues() : [];
  const devices = regRows.filter(r => String(r[regCol['School Name']] || '').trim() === schoolName);

  if (devices.length === 0) return { ok: false, error: 'no_devices', school: schoolName };

  // Data rows within the period
  let rows = [];
  if (dataSheet) {
    const dLast = dataSheet.getLastRow();
    if (dLast >= 2) {
      const dHeaders = dataSheet.getRange(1, 1, 1, dataSheet.getLastColumn()).getValues()[0].map(String);
      const dCol = {}; dHeaders.forEach((h, i) => dCol[h.trim()] = i);
      const raw = dataSheet.getRange(2, 1, dLast - 1, dHeaders.length).getValues();
      raw.forEach(r => {
        const dt = r[dCol['DateTime']];
        const dtDate = (dt instanceof Date) ? dt : (dt ? new Date(dt) : null);
        if (!dtDate || isNaN(dtDate)) return;
        if (dtDate < range.start || dtDate > range.end) return;
        const obj = {};
        dHeaders.forEach((h, i) => obj[h.trim()] = r[i]);
        obj._dt = dtDate;
        rows.push(obj);
      });
    }
  }

  // Per-device aggregates
  const perDevice = devices.map(dev => {
    const deviceId = String(dev[regCol['Device ID']] || '').trim().toUpperCase();
    const classroom = String(dev[regCol['Active Classroom']] || dev[regCol['Set Classroom']] || 'Classroom').trim();
    const hollyName = String(dev[regCol['Holly Name']] || '').trim();
    const teacher   = String(dev[regCol['Teacher Name']] || '').trim();
    const yearGroup = String(dev[regCol['Set Year Group']] || dev[regCol['Active Year Group']] || '').trim();

    const devRows = rows.filter(r => String(r['Device ID'] || '').trim().toUpperCase() === deviceId);
    const sensorRows = devRows.filter(r => {
      const db = Number(r['Sound dB']);
      return !r['Event Type'] && !isNaN(db) && db >= 10 && db <= 120;
    });

    const states = { Green: 0, Amber: 0, Red: 0, Sleeping: 0, Other: 0 };
    sensorRows.forEach(r => {
      const s = String(r['LED State'] || '').trim();
      if (s === 'Green' || s === 'Reward (Quiet)') states.Green++;
      else if (s === 'Amber') states.Amber++;
      else if (s === 'Red') states.Red++;
      else if (s.indexOf('Sleep') >= 0) states.Sleeping++;
      else states.Other++;
    });
    const totalSensor = sensorRows.length;
    const pct = n => totalSensor > 0 ? Math.round(n / totalSensor * 1000) / 10 : 0;

    const dbVals = sensorRows.map(r => Number(r['Sound dB'])).filter(v => !isNaN(v));
    const meanDb = dbVals.length ? +(dbVals.reduce((a,b) => a + b, 0) / dbVals.length).toFixed(1) : null;
    const peakDb = dbVals.length ? +Math.max.apply(null, dbVals).toFixed(1) : null;

    const tempVals = sensorRows.map(r => Number(r['Temperature (°C)'])).filter(v => !isNaN(v) && v > 5 && v < 40);
    const meanTemp = tempVals.length ? +(tempVals.reduce((a,b) => a + b, 0) / tempVals.length).toFixed(1) : null;
    const peakTemp = tempVals.length ? +Math.max.apply(null, tempVals).toFixed(1) : null;

    const quietRewards = devRows.filter(r => String(r['Event Type'] || '').trim() === 'QUIET_REWARD' && String(r['Event'] || '').trim() === 'TRIGGER').length;
    const hollyHops    = devRows.filter(r => String(r['Event Type'] || '').trim() === 'HOLLY_HOP'     && String(r['Event'] || '').trim() === 'START').length;

    // Sustained red streaks (>=5 consecutive)
    const sortedSensor = sensorRows.slice().sort((a,b) => a._dt - b._dt);
    let streak = 0, sustainedCount = 0, longestStreak = 0, longestStart = null, curStart = null;
    sortedSensor.forEach(r => {
      const s = String(r['LED State'] || '').trim();
      if (s === 'Red') {
        if (streak === 0) curStart = r._dt;
        streak++;
        if (streak > longestStreak) { longestStreak = streak; longestStart = curStart; }
      } else {
        if (streak >= 5) sustainedCount++;
        streak = 0;
      }
    });
    if (streak >= 5) sustainedCount++;

    // Hour-of-day red buckets — find most Red hour
    const hourRed = {};
    sensorRows.forEach(r => {
      if (String(r['LED State'] || '').trim() === 'Red') {
        const h = r._dt.getHours();
        hourRed[h] = (hourRed[h] || 0) + 1;
      }
    });
    let worstHour = null, worstHourCount = 0;
    Object.keys(hourRed).forEach(h => { if (hourRed[h] > worstHourCount) { worstHour = Number(h); worstHourCount = hourRed[h]; }});

    return {
      device_id:     deviceId,
      classroom:     classroom,
      holly_name:    hollyName,
      teacher:       teacher,
      year_group:    yearGroup,
      rows_logged:   totalSensor,
      mean_db:       meanDb,
      peak_db:       peakDb,
      pct_green:     pct(states.Green),
      pct_amber:     pct(states.Amber),
      pct_red:       pct(states.Red),
      pct_sleeping:  pct(states.Sleeping),
      mean_temp_c:   meanTemp,
      peak_temp_c:   peakTemp,
      quiet_rewards: quietRewards,
      holly_hops:    hollyHops,
      sustained_red_episodes: sustainedCount,
      longest_red_streak_minutes: longestStreak,  // each row ~ 1 min
      longest_red_started:  longestStart ? Utilities.formatDate(longestStart, Session.getScriptTimeZone(), 'HH:mm on EEEE d MMM') : null,
      noisiest_hour: worstHour !== null ? String(worstHour).padStart(2,'0') + ':00' : null,
      coverage_days: new Set(sensorRows.map(r => Utilities.formatDate(r._dt, Session.getScriptTimeZone(), 'yyyy-MM-dd'))).size
    };
  });

  // School-level rollup
  const activeDevices = perDevice.filter(d => d.rows_logged > 0);
  const meanGreen = activeDevices.length ? +(activeDevices.reduce((s,d) => s + d.pct_green, 0) / activeDevices.length).toFixed(1) : 0;
  const meanRed   = activeDevices.length ? +(activeDevices.reduce((s,d) => s + d.pct_red,   0) / activeDevices.length).toFixed(1) : 0;
  const totalRewards = activeDevices.reduce((s,d) => s + d.quiet_rewards, 0);
  const totalHops    = activeDevices.reduce((s,d) => s + d.holly_hops, 0);
  const best  = activeDevices.slice().sort((a,b) => b.pct_green - a.pct_green)[0] || null;
  const worst = activeDevices.slice().sort((a,b) => a.pct_green - b.pct_green)[0] || null;

  return {
    ok: true,
    school: schoolName,
    period_type: periodType,
    period_key: range.key,
    period_start: Utilities.formatDate(range.start, Session.getScriptTimeZone(), 'yyyy-MM-dd'),
    period_end:   Utilities.formatDate(range.end,   Session.getScriptTimeZone(), 'yyyy-MM-dd'),
    devices_registered: devices.length,
    devices_with_data:  activeDevices.length,
    devices_offline:    devices.length - activeDevices.length,
    school_mean_green_pct: meanGreen,
    school_mean_red_pct:   meanRed,
    school_total_quiet_rewards: totalRewards,
    school_total_holly_hops:    totalHops,
    best_classroom:  best  ? { name: best.classroom,  pct_green: best.pct_green,  teacher: best.teacher }  : null,
    worst_classroom: worst ? { name: worst.classroom, pct_green: worst.pct_green, teacher: worst.teacher } : null,
    per_device: perDevice
  };
}


/* ==================== LLM CALL ==================== */

function callAnthropicForInsight(stats) {
  const apiKey = PropertiesService.getScriptProperties().getProperty('ANTHROPIC_API_KEY');
  if (!apiKey) throw new Error('ANTHROPIC_API_KEY not set in Script Properties');

  const userMsg = [
    "Report type: " + stats.period_type,
    "School: " + stats.school,
    "Period: " + stats.period_start + " to " + stats.period_end + " (" + stats.period_key + ")",
    "",
    "Here is the computed stats payload for this period:",
    "```json",
    JSON.stringify(stats, null, 2),
    "```",
    "",
    "Write Holly's insight for this " + stats.period_type + " report. Return only the JSON object in the format described."
  ].join("\n");

  const resp = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
    method: 'post',
    headers: {
      'x-api-key': apiKey,
      'anthropic-version': '2023-06-01',
      'content-type': 'application/json'
    },
    payload: JSON.stringify({
      model: INSIGHTS_MODEL,
      max_tokens: INSIGHTS_MAX_TOKENS,
      system: HOLLY_SYSTEM_PROMPT,
      messages: [{ role: 'user', content: userMsg }]
    }),
    muteHttpExceptions: true
  });

  const code = resp.getResponseCode();
  const body = resp.getContentText();
  if (code !== 200) throw new Error('Anthropic API ' + code + ': ' + body.slice(0, 400));

  const parsed = JSON.parse(body);
  const text = (parsed.content || []).map(b => b.text || '').join('').trim();

  // Strip any code fences the model may have added despite instructions
  const cleaned = text.replace(/^```json\s*/i, '').replace(/^```\s*/i, '').replace(/```\s*$/, '').trim();

  let insight;
  try { insight = JSON.parse(cleaned); }
  catch (e) { throw new Error('LLM returned non-JSON: ' + cleaned.slice(0, 400)); }

  return insight;
}


/* ==================== VALIDATION — no hallucinated numbers ==================== */

function validateInsight(insight, stats) {
  // Check shape
  if (!insight || typeof insight !== 'object') return 'not_an_object';
  if (!insight.headline || typeof insight.headline !== 'string') return 'missing_headline';
  if (!Array.isArray(insight.observations) || insight.observations.length !== 3) return 'wrong_observation_count';
  if (!insight.suggestion || typeof insight.suggestion !== 'string') return 'missing_suggestion';
  for (let i = 0; i < 3; i++) {
    const o = insight.observations[i];
    if (!o || !o.text || !o.tone) return 'malformed_observation_' + (i+1);
    if (['positive','neutral','caution'].indexOf(o.tone) < 0) return 'bad_tone_' + (i+1);
  }

  // Number hallucination check: every number-looking token in the output must appear in the stats
  const allText = [insight.headline, insight.suggestion, insight.caveat || '']
    .concat(insight.observations.map(o => o.text)).join(' ');
  const numbersInText = (allText.match(/\b\d+(?:\.\d+)?\b/g) || [])
    .filter(n => n.length >= 2 || Number(n) >= 4); // ignore "1" "2" "3" common in prose
  const statsStr = JSON.stringify(stats);
  const missing = numbersInText.filter(n => statsStr.indexOf(n) < 0);
  if (missing.length > 0) return 'hallucinated_numbers:' + missing.join(',');

  return null; // valid
}


/* ==================== ORCHESTRATION ==================== */

function generateInsightForSchool(schoolName, periodType, force) {
  try {
    const stats = computePeriodStats(schoolName, periodType);
    if (!stats.ok) return { ok: false, school: schoolName, period: periodType, error: stats.error };

    // Skip if already published for this period (unless forced)
    if (!force && findInsightRow(schoolName, periodType, stats.period_key)) {
      return { ok: true, school: schoolName, period: periodType, skipped: 'exists' };
    }

    const insight = callAnthropicForInsight(stats);
    const err = validateInsight(insight, stats);

    const insightId = makeInsightId(schoolName, periodType, stats.period_key);
    const status = err ? 'error' : 'published';

    writeInsightRow({
      id: insightId,
      school: schoolName,
      deviceId: '',
      scope: 'school',
      periodType: periodType,
      periodKey: stats.period_key,
      periodStart: stats.period_start,
      periodEnd:   stats.period_end,
      status: status,
      headline: insight.headline || '',
      observations: insight.observations || [],
      suggestion: insight.suggestion || '',
      caveat: insight.caveat || '',
      rawStats: JSON.stringify(stats),
      errorMessage: err || ''
    });

    return { ok: !err, school: schoolName, period: periodType, insight_id: insightId, validation_error: err || null };
  } catch (e) {
    console.error('generateInsightForSchool failed', schoolName, periodType, e);
    return { ok: false, school: schoolName, period: periodType, error: String(e) };
  }
}

function generateAllInsights() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: false, error: 'no_registry' };

  const regLast = regSheet.getLastRow();
  if (regLast < 2) return { ok: true, processed: 0 };

  const regHeaders = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0].map(String);
  const schoolCol = regHeaders.indexOf('School Name');
  const allSchools = Array.from(new Set(
    regSheet.getRange(2, schoolCol + 1, regLast - 1, 1).getValues().flat().map(s => String(s || '').trim()).filter(Boolean)
  )).sort();

  // Determine which periods to check today. Weekly every Mon, monthly on 1st, etc.
  const today = new Date();
  const dow = today.getDay();   // 0=Sun, 1=Mon
  const dom = today.getDate();
  const mo  = today.getMonth(); // 0=Jan

  const periods = ['monthly']; // always try monthly — it's the flagship
  if (dow === 1) periods.unshift('weekly');
  if (dom === 1) periods.push('halfterm');
  if (dom === 1 && (mo === 0 || mo === 3 || mo === 7)) periods.push('termly'); // approx term boundaries
  if (dom === 1 && mo === 8) periods.push('yearly');

  // Batching — cursor
  const props = PropertiesService.getScriptProperties();
  let cursor = Number(props.getProperty(INSIGHTS_CURSOR_KEY) || 0);
  if (cursor >= allSchools.length) cursor = 0;

  const batch = allSchools.slice(cursor, cursor + INSIGHTS_BATCH_LIMIT);
  const results = [];
  batch.forEach(school => {
    periods.forEach(p => {
      results.push(generateInsightForSchool(school, p, false));
    });
  });

  cursor += batch.length;
  if (cursor >= allSchools.length) cursor = 0;
  props.setProperty(INSIGHTS_CURSOR_KEY, String(cursor));

  return {
    ok: true,
    processed: batch.length,
    total_schools: allSchools.length,
    periods_checked: periods,
    next_cursor: cursor,
    results: results
  };
}

function readInsightsForSchool(schoolName, periodType) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(INSIGHTS_SHEET);
  if (!sheet) return { ok: true, school: schoolName, insights: [] };
  const last = sheet.getLastRow();
  if (last < 2) return { ok: true, school: schoolName, insights: [] };

  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(String);
  const c = {}; headers.forEach((h, i) => c[h.trim()] = i);
  const data = sheet.getRange(2, 1, last - 1, headers.length).getValues();

  const out = [];
  data.forEach(r => {
    if (String(r[c['School Name']] || '').trim() !== schoolName) return;
    if (periodType && String(r[c['Period Type']] || '').trim() !== periodType) return;
    const status = String(r[c['Status']] || '').trim();
    if (status !== 'published' && status !== 'override') return;

    const override = String(r[c['Manual Override']] || '').trim();

    out.push({
      id:          r[c['Insight ID']],
      period_type: r[c['Period Type']],
      period_key:  r[c['Period Key']],
      period_start: formatDateString(r[c['Period Start']]),
      period_end:   formatDateString(r[c['Period End']]),
      generated_at: formatDateString(r[c['Generated At']], true),
      model:        r[c['Model']],
      prompt_version: r[c['Prompt Version']],
      headline:    override || String(r[c['Headline']] || ''),
      observations: [
        { tone: String(r[c['Observation 1 Tone']] || 'neutral'), text: String(r[c['Observation 1']] || '') },
        { tone: String(r[c['Observation 2 Tone']] || 'neutral'), text: String(r[c['Observation 2']] || '') },
        { tone: String(r[c['Observation 3 Tone']] || 'neutral'), text: String(r[c['Observation 3']] || '') }
      ],
      suggestion: String(r[c['Suggestion']] || ''),
      caveat:     String(r[c['Caveat']] || ''),
      has_override: !!override,
      feedback_up:   Number(r[c['Feedback Up']] || 0),
      feedback_down: Number(r[c['Feedback Down']] || 0)
    });
  });

  // Most recent per period_type only
  const byType = {};
  out.forEach(i => {
    if (!byType[i.period_type] || byType[i.period_type].period_key < i.period_key) {
      byType[i.period_type] = i;
    }
  });

  return { ok: true, school: schoolName, insights: Object.keys(byType).map(k => byType[k]) };
}

function recordInsightFeedback(data) {
  const id = String(data.insight_id || '').trim();
  const vote = String(data.vote || '').trim();
  if (!id || (vote !== 'up' && vote !== 'down')) return { ok: false, error: 'bad_request' };

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(INSIGHTS_SHEET);
  if (!sheet) return { ok: false, error: 'no_insights_tab' };
  const last = sheet.getLastRow();
  if (last < 2) return { ok: false, error: 'no_insights' };

  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(String);
  const c = {}; headers.forEach((h, i) => c[h.trim()] = i);
  const ids = sheet.getRange(2, c['Insight ID'] + 1, last - 1, 1).getValues();

  for (let i = 0; i < ids.length; i++) {
    if (String(ids[i][0] || '').trim() === id) {
      const colIdx = vote === 'up' ? c['Feedback Up'] : c['Feedback Down'];
      const cell = sheet.getRange(i + 2, colIdx + 1);
      cell.setValue((Number(cell.getValue()) || 0) + 1);
      return { ok: true, id: id, vote: vote };
    }
  }
  return { ok: false, error: 'insight_not_found' };
}

// Captures teacher sentiment about each day's classroom experience.
// One submission per teacher per day. Editable until midnight (we just
// overwrite the day's row for that user). Auto-captures the day's actual
// dashboard numbers so we can correlate sentiment with measured behaviour.

const FEEDBACK_SHEET = 'Daily Feedback';
const FEEDBACK_HEADERS = [
  'Submitted At',
  'Date',
  'Username',
  'School',
  'Classroom',
  'Device ID',
  'Rating',          // 'up' or 'down'
  'Focus Stars',     // 1-5
  'Comment',         // free text, optional
  'Score',           // captured from dashboard data at time of submission
  'Green %',
  'Red %',
  'Hops',
  'Rewards',
  'Edited Count'     // bumped each time they edit today's row
];

/**
 * Ensure the Daily Feedback sheet exists with headers.
 * Idempotent — safe to call on every submission.
 */
function ensureFeedbackSheet_() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sh = ss.getSheetByName(FEEDBACK_SHEET);
  if (!sh) {
    sh = ss.insertSheet(FEEDBACK_SHEET);
    sh.getRange(1, 1, 1, FEEDBACK_HEADERS.length).setValues([FEEDBACK_HEADERS]);
    sh.getRange(1, 1, 1, FEEDBACK_HEADERS.length).setFontWeight('bold').setBackground('#1C2444').setFontColor('#FFFFFF');
    sh.setFrozenRows(1);
    sh.setColumnWidth(1, 160);  // submitted at
    sh.setColumnWidth(9, 400);  // comment
  } else {
    // Add any missing headers without disturbing existing data
    const lastCol = sh.getLastColumn();
    if (lastCol > 0) {
      const existing = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
      let added = 0;
      FEEDBACK_HEADERS.forEach(h => {
        if (!existing.includes(h)) {
          sh.getRange(1, existing.length + added + 1).setValue(h).setFontWeight('bold');
          added++;
        }
      });
    }
  }
  return sh;
}

/**
 * Find a teacher's feedback row for today (if it exists).
 * Returns { rowIndex, headers, data } or null.
 */
function findFeedbackRowForToday_(sh, username, todayStr) {
  const last = sh.getLastRow();
  if (last < 2) return null;
  const lastCol = sh.getLastColumn();
  const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
  const c = {}; headers.forEach((h, i) => c[h] = i);

  const dateCol = c['Date'];
  const userCol = c['Username'];
  if (dateCol === undefined || userCol === undefined) return null;

  const data = sh.getRange(2, 1, last - 1, lastCol).getValues();
  for (let i = 0; i < data.length; i++) {
    const rowDate = data[i][dateCol];
    let rowDateStr = '';
    if (rowDate instanceof Date) {
      rowDateStr = Utilities.formatDate(rowDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
    } else {
      rowDateStr = String(rowDate || '').substring(0, 10);
    }
    const rowUser = String(data[i][userCol] || '').trim();
    if (rowDateStr === todayStr && rowUser === username) {
      return { rowIndex: i + 2, headers: headers, c: c, data: data[i] };
    }
  }
  return null;
}

/**
 * Handle POST action=submit_daily_feedback.
 * Expected: { token, rating ('up'|'down'), stars (1-5), comment, classroom,
 *             device_id, score, green_pct, red_pct, hops, rewards }
 * The dashboard captures the day's numbers from what's on screen and sends
 * them along — no need for the server to recompute them.
 */
function handleSubmitDailyFeedback_(data) {
  try {
    const auth = validateToken_(data.token);
    if (!auth.ok) return auth;

    // Validate inputs
    const rating = String(data.rating || '').trim().toLowerCase();
    if (rating !== 'up' && rating !== 'down') {
      return { ok: false, detail: 'Rating must be up or down.' };
    }
    const stars = Math.max(1, Math.min(5, Number(data.stars) || 0));
    if (!stars) {
      return { ok: false, detail: 'Please select 1-5 stars for focus.' };
    }
    let comment = String(data.comment || '').trim();
    if (comment.length > 1000) comment = comment.substring(0, 1000);

    const classroom = String(data.classroom || '').trim();
    const deviceId  = String(data.device_id || '').trim().toUpperCase();

    // Numeric snapshots — accept whatever the dashboard sends, default to ''
    const num = v => (v === '' || v === null || v === undefined || isNaN(Number(v))) ? '' : Number(v);
    const score    = num(data.score);
    const greenPct = num(data.green_pct);
    const redPct   = num(data.red_pct);
    const hops     = num(data.hops);
    const rewards  = num(data.rewards);

    const sh = ensureFeedbackSheet_();
    const tz = Session.getScriptTimeZone();
    const now = new Date();
    const todayStr = Utilities.formatDate(now, tz, 'yyyy-MM-dd');

    // Look up existing row — overwrite if found (editable until midnight),
    // append if not.
    const existing = findFeedbackRowForToday_(sh, auth.username, todayStr);
    const lastCol = sh.getLastColumn();
    const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
    const c = {}; headers.forEach((h, i) => c[h] = i);

    const row = new Array(headers.length).fill('');
    const set = (h, v) => { if (c[h] !== undefined) row[c[h]] = v; };

    set('Submitted At', now);
    set('Date',         todayStr);
    set('Username',     auth.username);
    set('School',       auth.school);
    set('Classroom',    classroom);
    set('Device ID',    deviceId);
    set('Rating',       rating);
    set('Focus Stars',  stars);
    set('Comment',      comment);
    set('Score',        score);
    set('Green %',      greenPct);
    set('Red %',        redPct);
    set('Hops',         hops);
    set('Rewards',      rewards);

    if (existing) {
      // Preserve original Submitted At, bump Edited Count
      const origSubmittedAt = existing.data[c['Submitted At']];
      if (origSubmittedAt) set('Submitted At', origSubmittedAt);
      const prevEdits = Number(existing.data[c['Edited Count']]) || 0;
      set('Edited Count', prevEdits + 1);
      sh.getRange(existing.rowIndex, 1, 1, headers.length).setValues([row]);
      return { ok: true, edited: true, edit_count: prevEdits + 1 };
    } else {
      set('Edited Count', 0);
      sh.appendRow(row);
      return { ok: true, edited: false };
    }
  } catch (err) {
    console.error('handleSubmitDailyFeedback_ error:', err);
    return { ok: false, detail: 'Could not save your feedback. Please try again.' };
  }
}

/**
 * Aggregated feedback summary for admin dashboard.
 * Returns counts and trends, never raw teacher names or comments unless
 * the caller has admin role (enforced in the doGet route, but defence in
 * depth here — never hand back named comments to non-admin).
 *
 * Returns:
 *   { ok, period_days, total_submissions, up_pct, avg_stars,
 *     by_day: [{date, up, down, avg_stars, n}],
 *     by_school: [{school, n, up_pct, avg_stars}]   (if no school filter),
 *     recent_comments: [{date, school, comment, rating}]  (admin-only) }
 */
function getFeedbackSummary_(schoolFilter, days) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sh = ss.getSheetByName(FEEDBACK_SHEET);
    if (!sh) return { ok: true, period_days: days, total_submissions: 0, by_day: [], by_school: [], recent_comments: [] };

    const last = sh.getLastRow();
    if (last < 2) return { ok: true, period_days: days, total_submissions: 0, by_day: [], by_school: [], recent_comments: [] };

    const tz = Session.getScriptTimeZone();
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - days);
    cutoff.setHours(0, 0, 0, 0);

    const lastCol = sh.getLastColumn();
    const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
    const c = {}; headers.forEach((h, i) => c[h] = i);
    const data = sh.getRange(2, 1, last - 1, lastCol).getValues();

    let upCount = 0, downCount = 0, starsSum = 0, starsN = 0;
    const byDay = {};      // date -> {up, down, starsSum, starsN}
    const bySchool = {};   // school -> {up, down, starsSum, starsN, n}
    const recentComments = [];

    data.forEach(r => {
      const dateVal = r[c['Date']];
      let dateStr = '';
      let dateObj = null;
      if (dateVal instanceof Date) {
        dateObj = dateVal;
        dateStr = Utilities.formatDate(dateObj, tz, 'yyyy-MM-dd');
      } else {
        dateStr = String(dateVal || '').substring(0, 10);
        if (/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) dateObj = new Date(dateStr);
      }
      if (!dateObj || dateObj < cutoff) return;

      const school = String(r[c['School']] || '').trim();
      if (schoolFilter && school !== schoolFilter) return;

      const rating = String(r[c['Rating']] || '').toLowerCase();
      const stars  = Number(r[c['Focus Stars']]) || 0;
      const comment = String(r[c['Comment']] || '').trim();

      if (rating === 'up') upCount++;
      else if (rating === 'down') downCount++;
      if (stars > 0) { starsSum += stars; starsN++; }

      if (!byDay[dateStr]) byDay[dateStr] = { up: 0, down: 0, starsSum: 0, starsN: 0 };
      if (rating === 'up')   byDay[dateStr].up++;
      if (rating === 'down') byDay[dateStr].down++;
      if (stars > 0) { byDay[dateStr].starsSum += stars; byDay[dateStr].starsN++; }

      if (!bySchool[school]) bySchool[school] = { up: 0, down: 0, starsSum: 0, starsN: 0, n: 0 };
      bySchool[school].n++;
      if (rating === 'up')   bySchool[school].up++;
      if (rating === 'down') bySchool[school].down++;
      if (stars > 0) { bySchool[school].starsSum += stars; bySchool[school].starsN++; }

      if (comment) {
        recentComments.push({
          date:    dateStr,
          school:  school,
          rating:  rating,
          stars:   stars,
          comment: comment
        });
      }
    });

    const total = upCount + downCount;
    const upPct = total > 0 ? Math.round(upCount / total * 1000) / 10 : 0;
    const avgStars = starsN > 0 ? Math.round(starsSum / starsN * 10) / 10 : 0;

    const byDayArr = Object.keys(byDay).sort().map(d => {
      const b = byDay[d];
      const t = b.up + b.down;
      return {
        date:      d,
        up:        b.up,
        down:      b.down,
        n:         t,
        up_pct:    t > 0 ? Math.round(b.up / t * 1000) / 10 : 0,
        avg_stars: b.starsN > 0 ? Math.round(b.starsSum / b.starsN * 10) / 10 : 0
      };
    });

    const bySchoolArr = Object.keys(bySchool).sort().map(s => {
      const b = bySchool[s];
      const t = b.up + b.down;
      return {
        school:    s,
        n:         b.n,
        up_pct:    t > 0 ? Math.round(b.up / t * 1000) / 10 : 0,
        avg_stars: b.starsN > 0 ? Math.round(b.starsSum / b.starsN * 10) / 10 : 0
      };
    });

    // Sort recent comments newest first, cap at 50
    recentComments.sort((a, b) => a.date < b.date ? 1 : -1);
    const cappedComments = recentComments.slice(0, 50);

    return {
      ok: true,
      period_days: days,
      total_submissions: total,
      up_count: upCount,
      down_count: downCount,
      up_pct: upPct,
      avg_stars: avgStars,
      by_day: byDayArr,
      by_school: bySchoolArr,
      recent_comments: cappedComments
    };
  } catch (err) {
    console.error('getFeedbackSummary_ error:', err);
    return { ok: false, detail: 'Could not load feedback summary.' };
  }
}

/**
 * Compute the pilot phase for a given submission date.
 * Returns { phase: 'baseline'|'mid'|'end'|'in-pilot'|'pre-pilot', day: <int> }.
 * day is days since PILOT_START_DATE (0 = pilot day 1; negative = before pilot).
 */
function getPilotPhase_(submittedAt) {
  const start = new Date(PILOT_START_DATE_STR + 'T00:00:00');
  const sub = (submittedAt instanceof Date) ? submittedAt : new Date(submittedAt);
  // Days difference, rounded down. Both dates normalised to local midnight.
  const startMid = new Date(start.getFullYear(), start.getMonth(), start.getDate());
  const subMid   = new Date(sub.getFullYear(),   sub.getMonth(),   sub.getDate());
  const day = Math.floor((subMid - startMid) / 86400000);

  let phase;
  if (day < 0)                                         phase = 'pre-pilot';
  else if (day < PILOT_BASELINE_END_DAYS)              phase = 'baseline';
  else if (day >= PILOT_MID_START_DAYS && day <= PILOT_MID_END_DAYS) phase = 'mid';
  else if (day >= PILOT_END_START_DAYS)                phase = 'end';
  else                                                 phase = 'in-pilot';

  return { phase: phase, day: day };
}

/**
 * Coerce a star value to integer 0-5. 0 = unanswered.
 */
function _starsOrZero_(v) {
  const n = Number(v);
  if (isNaN(n)) return 0;
  return Math.max(0, Math.min(5, Math.round(n)));
}

/**
 * Handle POST action=submit_full_review.
 * Expected payload: { token, nps, dim_<each>, comment }
 * All dimensions are optional individually but at least ONE rating
 * (any star or NPS) AND a comment OR a non-zero number of dimensions
 * must be present — otherwise it's an empty submission.
 */
function handleSubmitFullReview_(data) {
  try {
    const auth = validateToken_(data.token);
    if (!auth.ok) return auth;

    // NPS — 0-10, optional
    let nps = Number(data.nps);
    if (isNaN(nps) || nps < 0 || nps > 10) nps = '';

    // Eight dimensions
    const dimValues = {};
    let dimsAnswered = 0;
    REVIEW_DIMENSIONS.forEach(d => {
      const v = _starsOrZero_(data[d.key]);
      dimValues[d.key] = v;
      if (v > 0) dimsAnswered++;
    });

    // Comment, capped
    let comment = String(data.comment || '').trim();
    if (comment.length > 5000) comment = comment.substring(0, 5000);

    // Reject empty submissions. Allow:
    //   - any dimension answered, OR
    //   - NPS set + a comment, OR
    //   - just a comment with a long enough body to be useful
    const hasNps = nps !== '';
    const hasDim = dimsAnswered > 0;
    if (!hasDim && !hasNps && !comment) {
      return { ok: false, detail: 'Please rate at least one dimension or leave a comment.' };
    }

    const sh = ensureReviewsSheet_();
    const lastCol = sh.getLastColumn();
    const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
    const c = {}; headers.forEach((h, i) => c[h] = i);

    const submittedAt = new Date();
    const phase = getPilotPhase_(submittedAt);

    const row = new Array(headers.length).fill('');
    const set = (h, v) => { if (c[h] !== undefined) row[c[h]] = v; };

    set('Submitted At', submittedAt);
    set('Pilot Phase',  phase.phase);
    set('Pilot Day',    phase.day);
    set('Username',     auth.username);
    set('School',       auth.school);
    set('Role',         auth.role);
    set('NPS',          nps);

    REVIEW_DIMENSIONS.forEach(d => {
      set(d.header, dimValues[d.key] || '');
    });

    set('Comment', comment);

    sh.appendRow(row);
    return { ok: true, phase: phase.phase, day: phase.day };
  } catch (err) {
    console.error('handleSubmitFullReview_ error:', err);
    return { ok: false, detail: 'Could not save your review. Please try again.' };
  }
}

/**
 * Return the caller's own past reviews, newest first.
 */
function getMyReviews_(auth, limit) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sh = ss.getSheetByName(REVIEWS_SHEET);
    if (!sh) return { ok: true, reviews: [] };
    const last = sh.getLastRow();

    const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
    const c = {}; headers.forEach((h, i) => c[h] = i);
    const data = sh.getRange(2, 1, last - 1, lastCol).getValues();

    const tz = Session.getScriptTimeZone();
    const out = [];
    for (let i = data.length - 1; i >= 0; i--) {
      if (String(data[i][c['Username']] || '').trim() !== auth.username) continue;
      const submitted = data[i][c['Submitted At']];
      const submittedStr = (submitted instanceof Date)
        ? Utilities.formatDate(submitted, tz, 'yyyy-MM-dd HH:mm')
        : String(submitted || '');

      const dims = {};
      REVIEW_DIMENSIONS.forEach(d => {
        const idx = c[d.header];
        if (idx !== undefined) {
          const v = Number(data[i][idx]);
          dims[d.key] = isNaN(v) ? 0 : v;
        }
      });

      out.push({
        submitted_at: submittedStr,
        phase:        String(data[i][c['Pilot Phase']] || ''),
        day:          c['Pilot Day'] !== undefined ? Number(data[i][c['Pilot Day']]) : null,
        nps:          (data[i][c['NPS']] === '' || data[i][c['NPS']] === undefined) ? null : Number(data[i][c['NPS']]),
        dimensions:   dims,
        comment:      String(data[i][c['Comment']] || '')
      });
      if (out.length >= limit) break;
    }
    return { ok: true, reviews: out };

/**
 * Aggregated reviews summary for admin dashboard.
 * Returns dimension averages, NPS distribution, phase-bucketed comparison,
 * and full named-comment list. NEVER call from non-admin context — the
 * doGet route enforces this.
 */
function getReviewsSummary_(schoolFilter, days) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sh = ss.getSheetByName(REVIEWS_SHEET);
    if (!sh) return _emptyReviewsSummary_(days);
    const last = sh.getLastRow();
    if (last < 2) return _emptyReviewsSummary_(days);

    const tz = Session.getScriptTimeZone();
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - days);
    cutoff.setHours(0, 0, 0, 0);

    const lastCol = sh.getLastColumn();
    const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
    const c = {}; headers.forEach((h, i) => c[h] = i);
    const data = sh.getRange(2, 1, last - 1, lastCol).getValues();

    let promoters = 0, passives = 0, detractors = 0, npsN = 0;
    const dimSums = {}, dimCounts = {};
    REVIEW_DIMENSIONS.forEach(d => { dimSums[d.key] = 0; dimCounts[d.key] = 0; });
    const byPhase = { baseline: [], mid: [], end: [], 'in-pilot': [], 'pre-pilot': [] };
    const reviews = [];

    data.forEach(r => {
      const submitted = r[c['Submitted At']];
      let submittedDate = (submitted instanceof Date) ? submitted : new Date(submitted);
      if (!submittedDate || isNaN(submittedDate.getTime())) return;
      if (submittedDate < cutoff) return;

      const school = String(r[c['School']] || '').trim();
      if (schoolFilter && school !== schoolFilter) return;

      const phase = String(r[c['Pilot Phase']] || '').trim();
      const npsRaw = r[c['NPS']];
      const nps = (npsRaw === '' || npsRaw === null || npsRaw === undefined) ? null : Number(npsRaw);

      if (nps !== null && !isNaN(nps)) {
        if (nps >= 9)      promoters++;
        else if (nps >= 7) passives++;
        else                detractors++;
        npsN++;
      }

      const dims = {};
      REVIEW_DIMENSIONS.forEach(d => {
        const v = Number(r[c[d.header]]);
        if (!isNaN(v) && v > 0) {
          dimSums[d.key]   += v;
          dimCounts[d.key] += 1;
          dims[d.key] = v;
        } else {
          dims[d.key] = null;
        }
      });

      if (byPhase[phase]) byPhase[phase].push({ nps: nps, dims: dims });

      reviews.push({
        submitted_at: Utilities.formatDate(submittedDate, tz, 'yyyy-MM-dd HH:mm'),
        username:     String(r[c['Username']] || ''),
        school:       school,
        role:         String(r[c['Role']]     || ''),
        phase:        phase,
        day:          c['Pilot Day'] !== undefined ? Number(r[c['Pilot Day']]) : null,
        nps:          nps,
        dimensions:   dims,
        comment:      String(r[c['Comment']]  || '')
      });
    });

    reviews.sort((a, b) => a.submitted_at < b.submitted_at ? 1 : -1);

    // Dimension averages (1-5 scale) AND scaled to 1-10 for the framework
    const dimAvgs = {};
    REVIEW_DIMENSIONS.forEach(d => {
      const n = dimCounts[d.key];
      const avg5  = n > 0 ? dimSums[d.key] / n : null;
      const avg10 = avg5  !== null ? +(avg5 * 2).toFixed(1) : null;
      dimAvgs[d.key] = {
        label: d.label,
        n: n,
        avg_5: avg5 !== null ? +avg5.toFixed(2) : null,
        avg_10: avg10
      };
    });

    // NPS score (-100 to +100)
    const npsScore = npsN > 0 ? Math.round((promoters - detractors) / npsN * 100) : null;

    // Phase comparison (mid vs baseline vs end)
    function phaseSummary(rows) {
      if (!rows.length) return null;
      const npsVals = rows.map(r => r.nps).filter(v => v !== null && !isNaN(v));
      const dimAvgs = {};
      REVIEW_DIMENSIONS.forEach(d => {
        const vals = rows.map(r => r.dims[d.key]).filter(v => v !== null && !isNaN(v) && v > 0);
        dimAvgs[d.key] = vals.length > 0 ? +(vals.reduce((a,b) => a+b, 0) / vals.length).toFixed(2) : null;
      });
      return {
        n: rows.length,
        nps_avg: npsVals.length > 0 ? +(npsVals.reduce((a,b) => a+b, 0) / npsVals.length).toFixed(1) : null,
        dim_avgs: dimAvgs
      };
    }

    return {
      ok: true,
      period_days: days,
      total: reviews.length,
      nps: {
        score: npsScore,
        promoters: promoters,
        passives: passives,
        detractors: detractors,
        n: npsN
      },
      dimensions: dimAvgs,
      phases: {
        baseline: phaseSummary(byPhase['baseline']),
        mid:      phaseSummary(byPhase['mid']),
        end:      phaseSummary(byPhase['end']),
        in_pilot: phaseSummary(byPhase['in-pilot'])
      },
      pilot_start: PILOT_START_DATE_STR,
      reviews: reviews
    };
  } catch (err) {
    console.error('getReviewsSummary_ error:', err);
    return { ok: false, detail: 'Could not load reviews summary.' };
  }
}

function _emptyReviewsSummary_(days) {
  return {
    ok: true, period_days: days, total: 0,
    nps: { score: null, promoters: 0, passives: 0, detractors: 0, n: 0 },
    dimensions: {}, phases: { baseline: null, mid: null, end: null, in_pilot: null },
    pilot_start: PILOT_START_DATE_STR, reviews: []
  };
}

function handleDeviceEvent_(data) {
  try {
    const deviceId = String(data.device_id || '').trim().toUpperCase();
    if (!deviceId) return { ok: false, error: 'missing_device_id' };

    const eventType = String(data.event_type || '').trim();
    if (!eventType) return { ok: false, error: 'missing_event_type' };

    // Look up school from registry (best-effort).
    let school = '';
    try {
      const reg = lookupDevice(deviceId);
      if (reg && reg.schoolName) school = reg.schoolName;
    } catch (e) { /* ignore — log event anyway */ }

    const sh = ensureEventsSheet_();
    const tz = Session.getScriptTimeZone();
    let ts;
    if (data.timestamp && String(data.timestamp).trim() && String(data.timestamp).trim() !== 'TIME_UNAVAILABLE') {
      ts = data.timestamp;
    } else {
      ts = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd HH:mm:ss');
    }

    const lock = LockService.getPublicLock();
    lock.waitLock(10000);
    try {
      sh.appendRow([
        ts,
        deviceId,
        school,
        eventType,
        String(data.value || ''),
        String(data.detail || ''),
        String(data.firmware_version || ''),
        String(data.mac_address || '')
      ]);
    } finally {
      lock.releaseLock();
    }

    return { ok: true };
  } catch (err) {
    console.error('handleDeviceEvent_ error', err);
    return { ok: false, error: 'event_failed', detail: String(err) };
  }
}

/* ===========================================================================
   v49: Fleet-Wide Mic Offset Auto-Correction
   ===========================================================================
   Pairs with Holly firmware v5077. Solves the cross-fleet mic sensitivity
   variance problem by computing a per-school median mic floor and queueing
   per-device offset corrections via the registry.

   Logic:
     1. Gather all devices with mic_floor_valid='Yes' AND mic_floor in
        plausible range (5–60 dB — excludes contaminated readings like
        Lyppard's 5RT @ 75.7).
     2. Group by school. For each school with >=2 valid devices:
        - Compute median of valid floors
        - For each device: offset = school_median - device_floor
        - If |offset| in [5, 25] dB AND not already pending/applied:
            queue it via 'Set Mic DB Offset Correction'
     3. Single-Holly schools: skipped (no peers to compare against).
     4. |offset| > 25 dB: skipped + flagged in script properties (likely
        broken mic, needs admin attention not auto-correction).
     5. |offset| < 5 dB: skipped (within normal unit variance, no
        correction needed).

   Triggered:
     - Manually via admin endpoint ?action=compute_mic_offsets
     - Automatically at the end of updateDeviceHealth when a new
       mic_floor_valid='Yes' is captured (rate-limited to once per
       10 minutes per school via script properties cache).
   =========================================================================== */

const MIC_OFFSET_MIN_ABS_DB = 5.0;       // smaller deltas treated as noise
const MIC_OFFSET_MAX_ABS_DB = 25.0;      // larger suggests broken mic, skip
const MIC_OFFSET_MIN_FLOOR_DB = 5.0;     // valid range lower bound
const MIC_OFFSET_MAX_FLOOR_DB = 60.0;    // valid range upper bound (excludes contamination)
const MIC_OFFSET_MIN_FLEET_SIZE = 2;     // need at least 2 units per school
const MIC_OFFSET_RATE_LIMIT_MS = 600000; // 10 min cooldown per school

/**
 * Compute per-school median mic floors and queue per-device corrections.
 * Returns a summary object suitable for admin inspection.
 *
 * Safe to call repeatedly — the function:
 *   - reads existing 'Set Mic DB Offset Correction' values to avoid
 *     re-queueing the same correction
 *   - rate-limits per school via script properties (default 10 min)
 *   - never queues a correction that's effectively the same as the
 *     currently-active one (within 0.5 dB)
 */
function computeFleetMicOffsets_(opts) {
  opts = opts || {};
  const force = opts.force === true;       // bypass rate limits (for admin button)
  const dryRun = opts.dryRun === true;     // compute but don't write

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: false, error: 'no_registry' };

  const lastRow = regSheet.getLastRow();
  const lastCol = regSheet.getLastColumn();
  if (lastRow < 2) return { ok: true, schools: 0, queued: 0, skipped: 0 };

  const headers = regSheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim());
  const data = regSheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

  // Resolve column indices once
  const colIdx = (name) => headers.indexOf(name);
  const cSchool        = colIdx('School Name');
  const cDeviceId      = colIdx('Device ID');
  const cMicFloor      = colIdx('Mic Floor');
  const cMicFloorValid = colIdx('Mic Floor Valid');
  const cActiveCorr    = colIdx('Active Mic DB Offset Correction');
  const cQueueCorr     = colIdx('Set Mic DB Offset Correction');

  if (cSchool < 0 || cDeviceId < 0 || cMicFloor < 0 || cMicFloorValid < 0) {
    return { ok: false, error: 'missing_columns', detail: 'core mic floor columns not found in registry' };
  }
  if (cActiveCorr < 0 || cQueueCorr < 0) {
    return { ok: false, error: 'missing_columns', detail: 'v49 columns missing — run header migration first' };
  }

  // Group devices by school, only keeping those with valid in-range floors.
  // v52: Devices with valid-flag=yes but floor outside [MIN, MAX] dB are no
  // longer silently dropped — they're flagged as out_of_band_floor hardware
  // faults so admins see them. They're still excluded from the median pool
  // (a device claiming 75 dB silence is broken, not just offset).
  const bySchool = {};
  const outOfBandFlags = [];
  data.forEach((row, idx) => {
    const school = String(row[cSchool] || '').trim();
    const deviceId = String(row[cDeviceId] || '').trim();
    if (!school || !deviceId) return;
    const valid = String(row[cMicFloorValid] || '').trim().toLowerCase() === 'yes';
    const floor = Number(row[cMicFloor]);
    if (!valid || !isFinite(floor)) return;
    if (floor < MIC_OFFSET_MIN_FLOOR_DB || floor > MIC_OFFSET_MAX_FLOOR_DB) {
      // Excluded from baseline, but flagged so admins can investigate.
      outOfBandFlags.push({
        deviceId: deviceId,
        school: school,
        reportedFloor: floor,
        reason: 'out_of_band_floor',
        validRange: [MIC_OFFSET_MIN_FLOOR_DB, MIC_OFFSET_MAX_FLOOR_DB]
      });
      return;
    }
    if (!bySchool[school]) bySchool[school] = [];
    bySchool[school].push({
      rowIndex: idx + 2,    // sheet row (1-based, header row 1)
      deviceId: deviceId,
      floor: floor,
      activeCorrection: Number(row[cActiveCorr]) || 0,
      queuedCorrection: row[cQueueCorr]  // raw cell, may be empty
    });
  });

  // Rate-limit cache (per-school)
  const props = PropertiesService.getScriptProperties();
  const now = Date.now();

  const summary = {
    ok: true,
    schools: 0,
    schoolsRateLimited: 0,
    schoolsTooSmall: 0,
    queued: 0,
    skippedSmallDelta: 0,
    skippedHardwareFault: 0,
    skippedSameAsActive: 0,
    skippedAlreadyPending: 0,
    detailsBySchool: {},
    flaggedHardware: []
  };

  // v52: Surface out-of-band floors found during grouping. These never enter
  // the median pool, but admins need to see them.
  outOfBandFlags.forEach(function(f) {
    summary.flaggedHardware.push(f);
    summary.skippedHardwareFault++;
  });

  Object.keys(bySchool).forEach(school => {
    summary.schools++;
    const units = bySchool[school];

    if (units.length < MIC_OFFSET_MIN_FLEET_SIZE) {
      summary.schoolsTooSmall++;
      summary.detailsBySchool[school] = { skipped: 'fleet_too_small', units: units.length };
      return;
    }

    // Rate limit
    const cacheKey = 'mic_offset_last_' + school.replace(/[^a-zA-Z0-9]/g, '_');
    if (!force) {
      const last = Number(props.getProperty(cacheKey) || 0);
      if (now - last < MIC_OFFSET_RATE_LIMIT_MS) {
        summary.schoolsRateLimited++;
        summary.detailsBySchool[school] = {
          skipped: 'rate_limited',
          minutesUntilNextRun: Math.ceil((MIC_OFFSET_RATE_LIMIT_MS - (now - last)) / 60000)
        };
        return;
      }
    }

    // Compute median floor
    const floors = units.map(u => u.floor).sort((a, b) => a - b);
    const mid = Math.floor(floors.length / 2);
    const median = floors.length % 2 === 0
      ? (floors[mid - 1] + floors[mid]) / 2
      : floors[mid];

    const schoolDetail = {
      median: Math.round(median * 10) / 10,
      units: units.length,
      decisions: []
    };

    units.forEach(u => {
      // The raw mic floor reported by Holly is AFTER her current correction is applied.
      // To compute what HER correction needs to be from the school's perspective,
      // we need the school median to be the target. So the new offset is:
      //   new_correction = (school_median - device_raw_floor) + already_applied_correction
      // But device_raw_floor = floor_reported / 1.0 (it's just the displayed floor)
      // Actually the floor is reported AFTER correction, so:
      //   device_unrcorrected_floor = device_reported_floor - active_correction
      //   target_correction = school_median - device_uncorrected_floor
      //                     = school_median - (device_reported_floor - active_correction)
      //                     = school_median - device_reported_floor + active_correction
      const uncorrected = u.floor - u.activeCorrection;
      const targetCorrection = median - uncorrected;
      const delta = targetCorrection - u.activeCorrection;  // change vs current

      const decision = {
        deviceId: u.deviceId,
        reportedFloor: Math.round(u.floor * 10) / 10,
        activeCorrection: Math.round(u.activeCorrection * 100) / 100,
        targetCorrection: Math.round(targetCorrection * 100) / 100,
        delta: Math.round(delta * 100) / 100
      };

      // Decide
      const absTarget = Math.abs(targetCorrection);
      const absDelta = Math.abs(delta);

      if (absTarget > MIC_OFFSET_MAX_ABS_DB) {
        decision.action = 'flag_hardware';
        summary.skippedHardwareFault++;
        summary.flaggedHardware.push({
          deviceId: u.deviceId, school: school,
          reportedFloor: u.floor, schoolMedian: median, suggestedCorrection: targetCorrection,
          reason: 'excessive_offset'
        });
      } else if (absTarget < MIC_OFFSET_MIN_ABS_DB) {
        decision.action = 'skip_small_delta';
        summary.skippedSmallDelta++;
      } else if (absDelta < 0.5) {
        decision.action = 'skip_same_as_active';
        summary.skippedSameAsActive++;
      } else if (u.queuedCorrection !== '' && u.queuedCorrection !== null && u.queuedCorrection !== undefined) {
        decision.action = 'skip_already_pending';
        summary.skippedAlreadyPending++;
      } else {
        decision.action = 'queue';
        if (!dryRun) {
          regSheet.getRange(u.rowIndex, cQueueCorr + 1).setValue(targetCorrection);
        }
        summary.queued++;
      }

      schoolDetail.decisions.push(decision);
    });

    summary.detailsBySchool[school] = schoolDetail;
    if (!dryRun) {
      props.setProperty(cacheKey, String(now));
    }
  });

  return summary;
}

/**
 * Public admin endpoint — manual trigger from the dashboard.
 * Requires admin token. Always runs with force=true.
 */
function handleComputeMicOffsets_(token) {
  const auth = validateToken_(token);
  if (!auth.ok) return auth;
  if (auth.role !== 'admin') return { ok: false, error: 'forbidden', detail: 'Admin only' };
  return computeFleetMicOffsets_({ force: true, dryRun: false });

/* ==================== v50: One-time school tab cleanup ==================== */
/**
 * Removes all HEARTBEAT rows from a school's data tab. Run ONCE per school
 * from the Apps Script editor after deploying v50, to recover speed on tabs
 * that have been growing under v49 (or earlier). Safe to re-run — idempotent.
 *
 * Usage in editor:
 *   cleanupSchoolTabHeartbeats('Lyppard Grange Primary School')
 *
 * This iterates from bottom-up (deleting rows shifts indices, so bottom-up
 * is the only safe direction) and uses a batch deleteRows() per contiguous
 * range to avoid the 100-call/min Apps Script API limit on large tabs.
 *
 * Holds a public lock so it cannot race with telemetry writes from devices.
 */
function cleanupSchoolTabHeartbeats(schoolName) {
  if (!schoolName) {
    return { ok: false, error: 'missing_school_name', detail: 'Pass school name string' };
  }

  const lock = LockService.getPublicLock();
  if (!lock.tryLock(60000)) {
    return { ok: false, error: 'lock_timeout', detail: 'Could not acquire lock — try again in a minute' };
  }

  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheetName = sanitizeSheetName(schoolName);
    const sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      return { ok: false, error: 'sheet_not_found', detail: 'No tab: ' + sheetName };
    }

    const lastRow = sheet.getLastRow();
    const lastCol = sheet.getLastColumn();
    if (lastRow < 2 || lastCol < 1) {
      return { ok: true, message: 'Tab empty — nothing to clean', removed: 0 };
    }

    const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function(h) { return String(h).trim(); });
    const evCol = headers.indexOf('Event Type');
    if (evCol < 0) {
      return { ok: false, error: 'no_event_type_col', detail: 'Tab missing Event Type column' };
    }

    // Read just the Event Type column for performance
    const evValues = sheet.getRange(2, evCol + 1, lastRow - 1, 1).getValues();

    // Identify rows to delete (1-indexed sheet row numbers, skipping header)
    const toDelete = [];
    for (var i = 0; i < evValues.length; i++) {
      const v = String(evValues[i][0] || '').toUpperCase().trim();
      if (v === 'HEARTBEAT') {
        toDelete.push(i + 2); // +2 = 1-indexed + skip header row
      }
    }

    if (toDelete.length === 0) {
      return { ok: true, message: 'No heartbeat rows found', removed: 0, totalScanned: lastRow - 1 };
    }

    // Batch contiguous ranges. Delete bottom-up (descending) so indices stay valid.
    toDelete.sort(function(a, b) { return b - a; }); // descending
    var batches = [];
    var batchStart = toDelete[0];
    var batchEnd = toDelete[0];
    for (var j = 1; j < toDelete.length; j++) {
      const r = toDelete[j];
      if (r === batchEnd - 1) {
        batchEnd = r; // contiguous, extend down
      } else {
        batches.push({ from: batchEnd, count: batchStart - batchEnd + 1 });
        batchStart = r;
        batchEnd = r;
      }
    }
    batches.push({ from: batchEnd, count: batchStart - batchEnd + 1 });

    // Delete each batch
    batches.forEach(function(b) {
      sheet.deleteRows(b.from, b.count);
    });

    SpreadsheetApp.flush();

    const removed = toDelete.length;
    const remaining = sheet.getLastRow() - 1; // minus header
    console.log('[cleanup] ' + sheetName + ': removed ' + removed + ' heartbeats, ' + remaining + ' rows remain');

    return {
      ok: true,
      message: 'Cleanup complete',
      sheet: sheetName,
      removed: removed,
      remaining: remaining,
      batches: batches.length
    };
  } catch (err) {
    console.error('cleanupSchoolTabHeartbeats error', err);
    return { ok: false, error: 'cleanup_failed', detail: String(err) };
  } finally {
    lock.releaseLock();
  }
}

    results.push({ school: school, result: r });
    console.log('[cleanupAll] ' + school + ': ' + JSON.stringify(r));

/* ================================================================
 * v51: DASHBOARD SUMMARY CACHE — pre-aggregated per-device-per-day
 * ================================================================
 *
 * Why this exists:
 *   buildDashboardData and buildDailyData were re-aggregating raw
 *   per-row telemetry on every request — 7,000+ rows × 15 devices ×
 *   nested filters = 25-30s response times that trip the dashboard's
 *   30s AbortController. Caching the response helped only briefly
 *   because Apps Script CacheService caps at 100KB per entry, and
 *   full payloads with 6-week history exceed that.
 *
 * How it works:
 *   1. SUMMARY_TAB ('Dashboard Cache') holds one row per device-per-day
 *      with all the aggregates the dashboard needs. ~5K rows per school
 *      per year — tiny, instant to query.
 *   2. A scheduled trigger (refreshDashboardCacheTrigger) runs every
 *      5 minutes, reads NEW rows from each school tab since last update,
 *      aggregates incrementally, upserts into SUMMARY_TAB.
 *   3. buildDashboardData / buildDailyData read from SUMMARY_TAB, not
 *      the raw school tab. Dashboard responses go 25s → under 1s.
 *
 * Lifecycle:
 *   - Run rebuildAllDashboardCache() ONCE from the script editor after
 *     deploying v51. Walks every existing school tab and populates the
 *     summary tab.
 *   - Install the trigger ONCE: installDashboardCacheTrigger().
 *   - From then on it's automatic.
 *
 * Trade-off:
 *   - Dashboard data is up to 5 min stale. Acceptable for behaviour
 *     dashboards. Teacher dashboards never need second-by-second.
 *   - The cache TIMESTAMP is included in the response so teachers see
 *     freshness ("Updated 2 min ago").
 */

const SUMMARY_TAB = 'Dashboard Cache';
const SUMMARY_HEADERS = [
  'Device ID', 'School', 'Date', 'Lesson Rows', 'Green Rows', 'Amber Rows',
  'Red Rows', 'Avg dB', 'Sum dB', 'Reward Count', 'Hop Count',
  'Spike Sum', 'Spike Count', 'Trans Sum', 'Trans Count',
  'Voice Sum', 'Voice Count', 'Last Updated'
];

/**
 * Ensures the summary tab exists with the right header row.
 * Idempotent — safe to call multiple times.
 */
function ensureSummaryTab_() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sh = ss.getSheetByName(SUMMARY_TAB);
  if (!sh) {
    sh = ss.insertSheet(SUMMARY_TAB);
    sh.appendRow(SUMMARY_HEADERS);
    sh.setFrozenRows(1);
    sh.getRange(1, 1, 1, SUMMARY_HEADERS.length).setFontWeight('bold');
  }
  return sh;
}

/**
 * Format a Date to YYYY-MM-DD in the script's timezone.
 * Pulled out as a helper so we can call it once per cache rebuild
 * instead of millions of times inside filter loops (which is what
 * was making buildDashboardData slow in the first place).
 */
function _dateKey(d, tz) {
  if (!(d instanceof Date)) {
    const parsed = new Date(d);
    if (isNaN(parsed)) return '';
    d = parsed;
  }
  return Utilities.formatDate(d, tz, 'yyyy-MM-dd');
}

/**
 * Aggregate a list of school-tab rows (objects keyed by header name)
 * into a Map<deviceId|date, aggregateRow>. Used by both rebuild and
 * incremental refresh paths.
 */
function _aggregateRows_(rows, schoolName, tz) {
  const agg = {}; // key = "DEVICE|DATE", value = aggregate object

  rows.forEach(function(r) {
    const evType = String(r['Event Type'] || '').toUpperCase().trim();
    // Skip heartbeats (defensive — they shouldn't exist post-v50)
    // and skip any row that's a HOP/REWARD-only event with no sound data.
    // Only include real lesson telemetry in the per-day aggregates.
    const sound = Number(r['Sound dB']);
    const hasSound = !isNaN(sound) && r['Sound dB'] !== '' && r['Sound dB'] !== null;

    const dev = String(r['Device ID'] || '').trim().toUpperCase();
    if (!dev) return;

    const dt = r['DateTime'] || r['Date'];
    const dateKey = _dateKey(dt, tz);
    if (!dateKey) return;

    const key = dev + '|' + dateKey;
    if (!agg[key]) {
      agg[key] = {
        deviceId: dev,
        school: schoolName,
        date: dateKey,
        lessonRows: 0, greenRows: 0, amberRows: 0, redRows: 0,
        sumDb: 0, rewardCount: 0, hopCount: 0,
        spikeSum: 0, spikeCount: 0,
        transSum: 0, transCount: 0,
        voiceSum: 0, voiceCount: 0
      };
    }
    const a = agg[key];

    // Hop and Reward events get their own counters (don't need sound to count)
    if (evType === 'HOLLY_HOP' && String(r['Event'] || '').trim().toUpperCase() === 'START') {
      a.hopCount++;
    }
    if (String(r['Quiet Reward'] || '').trim().toLowerCase() === 'yes') {
      a.rewardCount++;
    }

    // Sound aggregates only for rows that actually have sound data
    if (hasSound) {
      a.lessonRows++;
      a.sumDb += sound;
      const ledState = String(r['LED State'] || '').trim();
      if (ledState === 'Green' || ledState === 'Reward (Quiet)') a.greenRows++;
      else if (ledState === 'Amber' || ledState === 'Deep Amber') a.amberRows++;
      else if (ledState === 'Red') a.redRows++;

      const sp = Number(r['Spike Count']);
      if (!isNaN(sp) && sp > 0) { a.spikeSum += sp; a.spikeCount++; }
      const tr = Number(r['Transition Avg dB']);
      if (!isNaN(tr) && tr > 0) { a.transSum += tr; a.transCount++; }
      const vp = Number(r['Voice %']);
      if (!isNaN(vp) && vp > 0) { a.voiceSum += vp; a.voiceCount++; }
    }
  });

  return agg;
}

/**
 * Read entire contents of a school tab into an array of objects keyed
 * by header name. Returns [] if tab missing or empty.
 */
function _readSchoolTab_(schoolName) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheetName = sanitizeSheetName(schoolName);
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) return { rows: [], headers: [] };
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow < 2 || lastCol < 1) return { rows: [], headers: [] };

  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(function(h) { return String(h).trim(); });
  const data = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();
  const rows = data.map(function(row) {
    const obj = {};
    headers.forEach(function(h, j) { obj[h] = row[j]; });
    return obj;
  });
  return { rows: rows, headers: headers };
}

/**
 * Replace ALL summary rows for a given school. Used by rebuildAllDashboardCache.
 * Safe to call any time — fully idempotent.
 */
function rebuildDashboardCacheForSchool(schoolName) {
  if (!schoolName) return { ok: false, error: 'missing_school' };

  const lock = LockService.getPublicLock();
  if (!lock.tryLock(60000)) {
    return { ok: false, error: 'lock_timeout', detail: 'Could not acquire lock' };
  }

  try {
    const sumSh = ensureSummaryTab_();
    const tz = Session.getScriptTimeZone();
    const tabData = _readSchoolTab_(schoolName);
    if (tabData.rows.length === 0) {
      return { ok: true, school: schoolName, message: 'School tab empty', rowsAggregated: 0 };
    }

    // Aggregate every row in the tab
    const agg = _aggregateRows_(tabData.rows, schoolName, tz);
    const aggKeys = Object.keys(agg);

    // Read existing summary rows and remove this school's existing entries
    const sumLastRow = sumSh.getLastRow();
    if (sumLastRow > 1) {
      const sumData = sumSh.getRange(2, 1, sumLastRow - 1, SUMMARY_HEADERS.length).getValues();
      const schoolColIdx = SUMMARY_HEADERS.indexOf('School');
      // Find rows belonging to this school, delete bottom-up
      const toDelete = [];
      for (var i = 0; i < sumData.length; i++) {
        if (String(sumData[i][schoolColIdx] || '').trim() === schoolName) {
          toDelete.push(i + 2); // sheet row number (1-indexed + skip header)
        }
      }
      toDelete.sort(function(a, b) { return b - a; });
      // Batch contiguous ranges for deletion
      var batchStart = null, batchEnd = null;
      const deleteBatches = [];
      for (var k = 0; k < toDelete.length; k++) {
        if (batchStart === null) { batchStart = toDelete[k]; batchEnd = toDelete[k]; }
        else if (toDelete[k] === batchEnd - 1) { batchEnd = toDelete[k]; }
        else { deleteBatches.push({ from: batchEnd, count: batchStart - batchEnd + 1 }); batchStart = toDelete[k]; batchEnd = toDelete[k]; }
      }
      if (batchStart !== null) deleteBatches.push({ from: batchEnd, count: batchStart - batchEnd + 1 });
      deleteBatches.forEach(function(b) { sumSh.deleteRows(b.from, b.count); });
    }

    // Append new aggregate rows
    if (aggKeys.length > 0) {
      const now = new Date();
      const newRows = aggKeys.map(function(k) {
        const a = agg[k];
        const avgDb = a.lessonRows > 0 ? a.sumDb / a.lessonRows : 0;
        return [
          a.deviceId, a.school, a.date,
          a.lessonRows, a.greenRows, a.amberRows, a.redRows,
          Math.round(avgDb * 10) / 10, a.sumDb,
          a.rewardCount, a.hopCount,
          a.spikeSum, a.spikeCount,
          a.transSum, a.transCount,
          a.voiceSum, a.voiceCount,
          now
        ];
      });
      const startRow = sumSh.getLastRow() + 1;
      sumSh.getRange(startRow, 1, newRows.length, SUMMARY_HEADERS.length).setValues(newRows);
    }

    SpreadsheetApp.flush();
    // Invalidate any cached dashboard payload for this school
    try { CacheService.getScriptCache().remove('dash:' + sanitizeSheetName(schoolName)); } catch(e) {}

    return {
      ok: true,
      school: schoolName,
      rowsScanned: tabData.rows.length,
      summariesWritten: aggKeys.length
    };
  } catch (err) {
    console.error('rebuildDashboardCacheForSchool error', err);
    return { ok: false, error: 'rebuild_failed', detail: String(err) };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Walks every distinct school in the registry and rebuilds its summary.
 * Run ONCE from the script editor after deploying v51.
 */
function rebuildAllDashboardCache() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return { ok: false, error: 'no_registry' };
  const lastRow = regSheet.getLastRow();
  if (lastRow < 2) return { ok: true, message: 'Registry empty', rebuilt: [] };

  const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
  const snCol = headers.map(function(h) { return String(h).trim(); }).indexOf('School Name');
  if (snCol < 0) return { ok: false, error: 'no_school_name_col' };

  const names = regSheet.getRange(2, snCol + 1, lastRow - 1, 1).getValues().flat()
    .map(function(s) { return String(s).trim(); })
    .filter(function(s) { return s.length > 0; });
  const distinct = [];
  names.forEach(function(n) { if (distinct.indexOf(n) < 0) distinct.push(n); });

  ensureSummaryTab_();
  const results = [];
  distinct.forEach(function(school) {

    results.push({ school: school, result: r });
    console.log('[rebuildAll] ' + school + ': ' + JSON.stringify(r));
  });
  return { ok: true, rebuilt: results };
}

/**
 * Trigger function: refresh today's summary for every active school.
 * Designed to run every 5 minutes. Only re-aggregates today's rows
 * (not the entire history) to stay fast.
 *
 * Why "today only": new telemetry only ever lands today. Yesterday's
 * aggregates never change. So we just rebuild today's row per device,
 * leave older rows alone. Bounded work regardless of school tab size.
 */
function refreshDashboardCacheTrigger() {
  // v52: 4-minute time budget + resumable cursor. Apps Script triggers die
  // at 6 min, so without this guard ~30 schools (or one bloated tab) could
  // get killed mid-iteration with no resume state. We persist a cursor
  // (`dashboard_cache_cursor`) with the next school to start from; cleared
  // when a full pass completes within budget.
  const startMs = Date.now();
  const TIME_BUDGET_MS = 4 * 60 * 1000;

  const ss = SpreadsheetApp.openById(SHEET_ID);
  const regSheet = ss.getSheetByName(REGISTRY_SHEET);
  if (!regSheet) return;
  const lastRow = regSheet.getLastRow();
  if (lastRow < 2) return;

  const headers = regSheet.getRange(1, 1, 1, regSheet.getLastColumn()).getValues()[0];
  const snCol = headers.map(function(h) { return String(h).trim(); }).indexOf('School Name');
  if (snCol < 0) return;

  const names = regSheet.getRange(2, snCol + 1, lastRow - 1, 1).getValues().flat()
    .map(function(s) { return String(s).trim(); })
    .filter(function(s) { return s.length > 0; });
  const distinct = [];
  names.forEach(function(n) { if (distinct.indexOf(n) < 0) distinct.push(n); });

  const tz = Session.getScriptTimeZone();
  const todayKey = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd');

  const props = PropertiesService.getScriptProperties();
  const cursor = props.getProperty('dashboard_cache_cursor') || '';
  Logger.log('refreshDashboardCacheTrigger: cursor=%s, schools=%d', cursor || '<start>', distinct.length);

  // Resolve start index from cursor. If cursor points at a school no longer
  // in the list (e.g. removed since the last run), fall back to the start.
  let startIdx = 0;
  if (cursor) {
    const found = distinct.indexOf(cursor);
    startIdx = (found >= 0) ? found : 0;
  }

  const schools = distinct;
  for (var i = startIdx; i < schools.length; i++) {
    const school = schools[i];
    try {
      _refreshTodayForSchool_(school, todayKey, tz);
    } catch (e) {
      console.error('[refreshTrigger] ' + school + ' failed:', e);
    }

    if (Date.now() - startMs > TIME_BUDGET_MS) {
      const nextSchool = schools[i + 1] || '';
      props.setProperty('dashboard_cache_cursor', nextSchool);
      Logger.log('refreshDashboardCacheTrigger: time budget hit, resuming next run from %s', nextSchool || '<reset>');
      return;
    }
  }

  // Full pass completed within budget — clear the cursor.
  props.deleteProperty('dashboard_cache_cursor');
}

/**
 * Helper: refresh just today's summary rows for one school.
 */
function _refreshTodayForSchool_(schoolName, todayKey, tz) {
  const lock = LockService.getPublicLock();
  if (!lock.tryLock(15000)) return; // skip if locked — next 5-min cycle picks it up

  try {
    const tabData = _readSchoolTab_(schoolName);
    if (tabData.rows.length === 0) return;

    // Filter to today's rows only
    const todayRows = tabData.rows.filter(function(r) {
      const dt = r['DateTime'] || r['Date'];
      return _dateKey(dt, tz) === todayKey;
    });

    const agg = _aggregateRows_(todayRows, schoolName, tz);
    const aggKeys = Object.keys(agg);

    const sumSh = ensureSummaryTab_();
    const sumLastRow = sumSh.getLastRow();
    const schoolColIdx = SUMMARY_HEADERS.indexOf('School');
    const dateColIdx = SUMMARY_HEADERS.indexOf('Date');

    // Find existing summary rows for this school and today, mark for replacement
    const toDelete = [];
    if (sumLastRow > 1) {
      const sumData = sumSh.getRange(2, 1, sumLastRow - 1, SUMMARY_HEADERS.length).getValues();
      for (var i = 0; i < sumData.length; i++) {
        if (String(sumData[i][schoolColIdx] || '').trim() === schoolName &&
            String(sumData[i][dateColIdx] || '').trim() === todayKey) {
          toDelete.push(i + 2);
        }
      }
      toDelete.sort(function(a, b) { return b - a; });
      toDelete.forEach(function(rowNum) { sumSh.deleteRows(rowNum, 1); });
    }

    // Append today's freshly-aggregated rows
    if (aggKeys.length > 0) {
      const now = new Date();
      const newRows = aggKeys.map(function(k) {
        const a = agg[k];
        const avgDb = a.lessonRows > 0 ? a.sumDb / a.lessonRows : 0;
        return [
          a.deviceId, a.school, a.date,
          a.lessonRows, a.greenRows, a.amberRows, a.redRows,
          Math.round(avgDb * 10) / 10, a.sumDb,
          a.rewardCount, a.hopCount,
          a.spikeSum, a.spikeCount,
          a.transSum, a.transCount,
          a.voiceSum, a.voiceCount,
          now
        ];
      });
      const startRow = sumSh.getLastRow() + 1;
      sumSh.getRange(startRow, 1, newRows.length, SUMMARY_HEADERS.length).setValues(newRows);
    }
    SpreadsheetApp.flush();
    // Bust the dashboard cache for this school
    try { CacheService.getScriptCache().remove('dash:' + sanitizeSheetName(schoolName)); } catch(e) {}
  } finally {
    lock.releaseLock();
  }
}

/**
 * Install the 5-minute trigger. Run ONCE from the script editor after deploy.
 * Safe to re-run — first deletes any existing copy of this trigger.
 */
function installDashboardCacheTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(function(t) {
    if (t.getHandlerFunction() === 'refreshDashboardCacheTrigger') {
      ScriptApp.deleteTrigger(t);
    }
  });
  ScriptApp.newTrigger('refreshDashboardCacheTrigger')
    .timeBased()
    .everyMinutes(5)
    .create();
  return { ok: true, message: 'Trigger installed — runs every 5 minutes' };
}

/**
 * Read summary rows from SUMMARY_TAB for a given school.
 * Returns an array of aggregate row objects, keyed by header name.
 */
function _readSummaryForSchool_(schoolName) {
  const sumSh = ensureSummaryTab_();
  const lastRow = sumSh.getLastRow();
  if (lastRow < 2) return [];
  const data = sumSh.getRange(2, 1, lastRow - 1, SUMMARY_HEADERS.length).getValues();
  const out = [];
  data.forEach(function(row) {
    if (String(row[1] || '').trim() !== schoolName) return; // School col
    const obj = {};
    SUMMARY_HEADERS.forEach(function(h, j) { obj[h] = row[j]; });
    out.push(obj);
  });
  return out;
}
